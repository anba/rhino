18      /**
19       * Create a JavaScript function object from a Java method.
20       *
21       * <p>The <code>member</code> argument must be either a java.lang.reflect.Method
22       * or a java.lang.reflect.Constructor and must match one of two forms.<p>
23       *
24       * The first form is a member with zero or more parameters
25       * of the following types: Object, String, boolean, Scriptable,
26       * int, or double. The Long type is not supported
27       * because the double representation of a long (which is the
28       * EMCA-mandated storage type for Numbers) may lose precision.
29       * If the member is a Method, the return value must be void or one
30       * of the types allowed for parameters.<p>
31       *
32       * The runtime will perform appropriate conversions based
33       * upon the type of the parameter. A parameter type of
34       * Object specifies that no conversions are to be done. A parameter
35       * of type String will use Context.toString to convert arguments.
36       * Similarly, parameters of type double, boolean, and Scriptable
37       * will cause Context.toNumber, Context.toBoolean, and
38       * Context.toObject, respectively, to be called.<p>
39       *
40       * If the method is not static, the Java 'this' value will
41       * correspond to the JavaScript 'this' value. Any attempt
42       * to call the function with a 'this' value that is not
43       * of the right Java type will result in an error.<p>
44       *
45       * The second form is the variable arguments (or "varargs")
46       * form. If the FunctionObject will be used as a constructor,
47       * the member must have the following parameters
48       * <pre>
49       *      (Context cx, Object[] args, Function ctorObj,
50       *       boolean inNewExpr)</pre>
51       * and if it is a Method, be static and return an Object result.<p>
52       *
53       * Otherwise, if the FunctionObject will <i>not</i> be used to define a
54       * constructor, the member must be a static Method with parameters
55       * <pre>
56       *      (Context cx, Scriptable thisObj, Object[] args,
57       *       Function funObj) </pre>
58       * and an Object result.<p>
59       *
60       * When the function varargs form is called as part of a function call,
61       * the <code>args</code> parameter contains the
62       * arguments, with <code>thisObj</code>
63       * set to the JavaScript 'this' value. <code>funObj</code>
64       * is the function object for the invoked function.<p>
65       *
66       * When the constructor varargs form is called or invoked while evaluating
67       * a <code>new</code> expression, <code>args</code> contains the
68       * arguments, <code>ctorObj</code> refers to this FunctionObject, and
69       * <code>inNewExpr</code> is true if and only if  a <code>new</code>
70       * expression caused the call. This supports defining a function that
71       * has different behavior when called as a constructor than when
72       * invoked as a normal function call. (For example, the Boolean
73       * constructor, when called as a function,
74       * will convert to boolean rather than creating a new object.)<p>
75       *
76       * @param name the name of the function
77       * @param methodOrConstructor a java.lang.reflect.Method or a java.lang.reflect.Constructor
78       *                            that defines the object
79       * @param scope enclosing scope of function
80       * @see org.mozilla.javascript.Scriptable
81       */
82      public FunctionObject(String name, Member methodOrConstructor,
83                            Scriptable scope)
84      {
85          if (methodOrConstructor instanceof Constructor) {
86              member = new MemberBox((Constructor<?>) methodOrConstructor);
87              isStatic = true; // well, doesn't take a 'this'
88          } else {
89              member = new MemberBox((Method) methodOrConstructor);
90              isStatic = member.isStatic();
91          }
92          String methodName = member.getName();
93          this.functionName = name;
94          Class<?>[] types = member.argTypes;
95          int arity = types.length;
96          if (arity == 4 && (types[1].isArray() || types[2].isArray())) {
97              // Either variable args or an error.
98              if (types[1].isArray()) {
99                  if (!isStatic ||
100                     types[0] != ScriptRuntime.ContextClass ||
101                     types[1].getComponentType() != ScriptRuntime.ObjectClass ||
102                     types[2] != ScriptRuntime.FunctionClass ||
103                     types[3] != Boolean.TYPE)
104                 {
105                     throw Context.reportRuntimeError1(
106                         "msg.varargs.ctor", methodName);
107                 }
108                 parmsLength = VARARGS_CTOR;
109             } else {
110                 if (!isStatic ||
111                     types[0] != ScriptRuntime.ContextClass ||
112                     types[1] != ScriptRuntime.ScriptableClass ||
113                     types[2].getComponentType() != ScriptRuntime.ObjectClass ||
114                     types[3] != ScriptRuntime.FunctionClass)
115                 {
116                     throw Context.reportRuntimeError1(
117                         "msg.varargs.fun", methodName);
118                 }
119                 parmsLength = VARARGS_METHOD;
120             }
121         } else {
122             parmsLength = arity;
123             if (arity > 0) {
124                 typeTags = new byte[arity];
125                 for (int i = 0; i != arity; ++i) {
126                     int tag = getTypeTag(types[i]);
127                     if (tag == JAVA_UNSUPPORTED_TYPE) {
128                         throw Context.reportRuntimeError2(
129                             "msg.bad.parms", types[i].getName(), methodName);
130                     }
131                     typeTags[i] = (byte)tag;
132                 }
133             }
134         }
135 
136         if (member.isMethod()) {
137             Method method = member.method();
138             Class<?> returnType = method.getReturnType();
139             if (returnType == Void.TYPE) {
140                 hasVoidReturn = true;
141             } else {
142                 returnTypeTag = getTypeTag(returnType);
143             }
144         } else {
145             Class<?> ctorType = member.getDeclaringClass();
146             if (!ScriptRuntime.ScriptableClass.isAssignableFrom(ctorType)) {
147                 throw Context.reportRuntimeError1(
148                     "msg.bad.ctor.return", ctorType.getName());
149             }
150         }
151 
152         ScriptRuntime.setFunctionProtoAndParent(this, scope);
153     }

155     /**
156      * @return One of <tt>JAVA_*_TYPE</tt> constants to indicate desired type
157      *         or {@link #JAVA_UNSUPPORTED_TYPE} if the convertion is not
158      *         possible
159      */
160     public static int getTypeTag(Class<?> type)
161     {
162         if (type == ScriptRuntime.StringClass)
163             return JAVA_STRING_TYPE;
164         if (type == ScriptRuntime.IntegerClass || type == Integer.TYPE)
165             return JAVA_INT_TYPE;
166         if (type == ScriptRuntime.BooleanClass || type == Boolean.TYPE)
167             return JAVA_BOOLEAN_TYPE;
168         if (type == ScriptRuntime.DoubleClass || type == Double.TYPE)
169             return JAVA_DOUBLE_TYPE;
170         if (ScriptRuntime.ScriptableClass.isAssignableFrom(type))
171             return JAVA_SCRIPTABLE_TYPE;
172         if (type == ScriptRuntime.ObjectClass)
173             return JAVA_OBJECT_TYPE;
174 
175         // Note that the long type is not supported; see the javadoc for
176         // the constructor for this class
177 
178         return JAVA_UNSUPPORTED_TYPE;
179     }

181     public static Object convertArg(Context cx, Scriptable scope,
182                                     Object arg, int typeTag)
183     {
184         switch (typeTag) {
185           case JAVA_STRING_TYPE:
186               if (arg instanceof String)
187                 return arg;
188             return ScriptRuntime.toString(arg);
189           case JAVA_INT_TYPE:
190               if (arg instanceof Integer)
191                 return arg;
192             return Integer.valueOf(ScriptRuntime.toInt32(arg));
193           case JAVA_BOOLEAN_TYPE:
194               if (arg instanceof Boolean)
195                 return arg;
196             return ScriptRuntime.toBoolean(arg) ? Boolean.TRUE
197                                                 : Boolean.FALSE;
198           case JAVA_DOUBLE_TYPE:
199             if (arg instanceof Double)
200                 return arg;
201             return new Double(ScriptRuntime.toNumber(arg));
202           case JAVA_SCRIPTABLE_TYPE:
203               return ScriptRuntime.toObjectOrNull(cx, arg, scope);
204           case JAVA_OBJECT_TYPE:
205             return arg;
206           default:
207             throw new IllegalArgumentException();
208         }
209     }

247     static Method findSingleMethod(Method[] methods, String name)
248     {
249         Method found = null;
250         for (int i = 0, N = methods.length; i != N; ++i) {
251             Method method = methods[i];
252             if (method != null && name.equals(method.getName())) {
253                 if (found != null) {
254                     throw Context.reportRuntimeError2(
255                         "msg.no.overload", name,
256                         method.getDeclaringClass().getName());
257                 }
258                 found = method;
259             }
260         }
261         return found;
262     }

264     /**
265      * Returns all public methods declared by the specified class. This excludes
266      * inherited methods.
267      *
268      * @param clazz the class from which to pull public declared methods
269      * @return the public methods declared in the specified class
270      * @see Class#getDeclaredMethods()
271      */
272     static Method[] getMethodList(Class<?> clazz) {
273         Method[] methods = null;
274         try {
275             // getDeclaredMethods may be rejected by the security manager
276             // but getMethods is more expensive
277             if (!sawSecurityException)
278                 methods = clazz.getDeclaredMethods();
279         } catch (SecurityException e) {
280             // If we get an exception once, give up on getDeclaredMethods
281             sawSecurityException = true;
282         }
283         if (methods == null) {
284             methods = clazz.getMethods();
285         }
286         int count = 0;
287         for (int i=0; i < methods.length; i++) {
288             if (sawSecurityException
289                 ? methods[i].getDeclaringClass() != clazz
290                 : !Modifier.isPublic(methods[i].getModifiers()))
291             {
292                 methods[i] = null;
293             } else {
294                 count++;
295             }
296         }
297         Method[] result = new Method[count];
298         int j=0;
299         for (int i=0; i < methods.length; i++) {
300             if (methods[i] != null)
301                 result[j++] = methods[i];
302         }
303         return result;
304     }

307      * Define this function as a JavaScript constructor.
308      * <p>
309      * Sets up the "prototype" and "constructor" properties. Also
310      * calls setParent and setPrototype with appropriate values.
311      * Then adds the function object as a property of the given scope, using
312      *      <code>prototype.getClassName()</code>
313      * as the name of the property.
314      *
315      * @param scope the scope in which to define the constructor (typically
316      *              the global object)
317      * @param prototype the prototype object
318      * @see org.mozilla.javascript.Scriptable#setParentScope
319      * @see org.mozilla.javascript.Scriptable#setPrototype
320      * @see org.mozilla.javascript.Scriptable#getClassName
321      */
322     public void addAsConstructor(Scriptable scope, Scriptable prototype)
323     {
324         initAsConstructor(scope, prototype);
325         defineProperty(scope, prototype.getClassName(),
326                        this, ScriptableObject.DONTENUM);
327     }

329     void initAsConstructor(Scriptable scope, Scriptable prototype)
330     {
331         ScriptRuntime.setFunctionProtoAndParent(this, scope);
332         setImmunePrototypeProperty(prototype);
333 
334         prototype.setParentScope(this);
335 
336         defineProperty(prototype, "constructor", this,
337                        ScriptableObject.DONTENUM  |
338                        ScriptableObject.PERMANENT |
339                        ScriptableObject.READONLY);
340         setParentScope(scope);
341     }

359     /**
360      * Performs conversions on argument types if needed and
361      * invokes the underlying Java method or constructor.
362      * <p>
363      * Implements Function.call.
364      *
365      * @see org.mozilla.javascript.Function#call(
366      *          Context, Scriptable, Scriptable, Object[])
367      */
368     @Override
369     public Object call(Context cx, Scriptable scope, Scriptable thisObj,
370                        Object[] args)
371     {
372         Object result;
373         boolean checkMethodResult = false;
374         int argsLength = args.length;
375 
376         for (int i = 0; i < argsLength; i++) {
377             // flatten cons-strings before passing them as arguments
378             if (args[i] instanceof ConsString) {
379                 args[i] = args[i].toString();
380             }
381         }
382 
383         if (parmsLength < 0) {
384             if (parmsLength == VARARGS_METHOD) {
385                 Object[] invokeArgs = { cx, thisObj, args, this };
386                 result = member.invoke(null, invokeArgs);
387                 checkMethodResult = true;
388             } else {
389                 boolean inNewExpr = (thisObj == null);
390                 Boolean b = inNewExpr ? Boolean.TRUE : Boolean.FALSE;
391                 Object[] invokeArgs = { cx, args, this, b };
392                 result = (member.isCtor())
393                          ? member.newInstance(invokeArgs)
394                          : member.invoke(null, invokeArgs);
395             }
396 
397         } else {
398             if (!isStatic) {
399                 Class<?> clazz = member.getDeclaringClass();
400                 if (!clazz.isInstance(thisObj)) {
401                     boolean compatible = false;
402                     if (thisObj == scope) {
403                         Scriptable parentScope = getParentScope();
404                         if (scope != parentScope) {
405                             // Call with dynamic scope for standalone function,
406                             // use parentScope as thisObj
407                             compatible = clazz.isInstance(parentScope);
408                             if (compatible) {
409                                 thisObj = parentScope;
410                             }
411                         }
412                     }
413                     if (!compatible) {
414                         // Couldn't find an object to call this on.
415                         throw ScriptRuntime.typeError1("msg.incompat.call",
416                                                        functionName);
417                     }
418                 }
419             }
420 
421             Object[] invokeArgs;
422             if (parmsLength == argsLength) {
423                 // Do not allocate new argument array if java arguments are
424                 // the same as the original js ones.
425                 invokeArgs = args;
426                 for (int i = 0; i != parmsLength; ++i) {
427                     Object arg = args[i];
428                     Object converted = convertArg(cx, scope, arg, typeTags[i]);
429                     if (arg != converted) {
430                         if (invokeArgs == args) {
431                             invokeArgs = args.clone();
432                         }
433                         invokeArgs[i] = converted;
434                     }
435                 }
436             } else if (parmsLength == 0) {
437                 invokeArgs = ScriptRuntime.emptyArgs;
438             } else {
439                 invokeArgs = new Object[parmsLength];
440                 for (int i = 0; i != parmsLength; ++i) {
441                     Object arg = (i < argsLength)
442                                  ? args[i]
443                                  : Undefined.instance;
444                     invokeArgs[i] = convertArg(cx, scope, arg, typeTags[i]);
445                 }
446             }
447 
448             if (member.isMethod()) {
449                 result = member.invoke(thisObj, invokeArgs);
450                 checkMethodResult = true;
451             } else {
452                 result = member.newInstance(invokeArgs);
453             }
454 
455         }
456 
457         if (checkMethodResult) {
458             if (hasVoidReturn) {
459                 result = Undefined.instance;
460             } else if (returnTypeTag == JAVA_UNSUPPORTED_TYPE) {
461                 result = cx.getWrapFactory().wrap(cx, scope, result, null);
462             }
463             // XXX: the code assumes that if returnTypeTag == JAVA_OBJECT_TYPE
464             // then the Java method did a proper job of converting the
465             // result to JS primitive or Scriptable to avoid
466             // potentially costly Context.javaToJS call.
467         }
468 
469         return result;
470     }

472     /**
473      * Return new {@link Scriptable} instance using the default
474      * constructor for the class of the underlying Java method.
475      * Return null to indicate that the call method should be used to create
476      * new objects.
477      */
478     @Override
479     public Scriptable createObject(Context cx, Scriptable scope) {
480         if (member.isCtor() || parmsLength == VARARGS_CTOR) {
481             return null;
482         }
483         Scriptable result;
484         try {
485             result = (Scriptable) member.getDeclaringClass().newInstance();
486         } catch (Exception ex) {
487             throw Context.throwAsScriptRuntimeEx(ex);
488         }
489 
490         result.setPrototype(getClassPrototype());
491         result.setParentScope(getParentScope());
492         return result;
493     }

503     private void readObject(ObjectInputStream in)
504         throws IOException, ClassNotFoundException
505     {
506         in.defaultReadObject();
507         if (parmsLength > 0) {
508             Class<?>[] types = member.argTypes;
509             typeTags = new byte[parmsLength];
510             for (int i = 0; i != parmsLength; ++i) {
511                 typeTags[i] = (byte)getTypeTag(types[i]);
512             }
513         }
514         if (member.isMethod()) {
515             Method method = member.method();
516             Class<?> returnType = method.getReturnType();
517             if (returnType == Void.TYPE) {
518                 hasVoidReturn = true;
519             } else {
520                 returnTypeTag = getTypeTag(returnType);
521             }
522         }
523     }

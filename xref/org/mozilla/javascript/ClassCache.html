13  /**
14   * Cache of generated classes and data structures to access Java runtime
15   * from JavaScript.
16   *
17   * @author Igor Bukanov
18   *
19   * @since Rhino 1.5 Release 5
20   */
21  public class ClassCache implements Serializable
22  {
23      private static final long serialVersionUID = -8866246036237312215L;
24      private static final Object AKEY = "ClassCache";
25      private volatile boolean cachingIsEnabled = true;
26      private transient HashMap<Class<?>,JavaMembers> classTable;
27      private transient HashMap<JavaAdapter.JavaAdapterSignature,Class<?>> classAdapterCache;
28      private transient HashMap<Class<?>,Object> interfaceAdapterCache;
29      private int generatedClassSerial;
30      private Scriptable associatedScope;

32      /**
33       * Search for ClassCache object in the given scope.
34       * The method first calls
35       * {@link ScriptableObject#getTopLevelScope(Scriptable scope)}
36       * to get the top most scope and then tries to locate associated
37       * ClassCache object in the prototype chain of the top scope.
38       *
39       * @param scope scope to search for ClassCache object.
40       * @return previously associated ClassCache object or a new instance of
41       *         ClassCache if no ClassCache object was found.
42       *
43       * @see #associate(ScriptableObject topScope)
44       */
45      public static ClassCache get(Scriptable scope)
46      {
47          ClassCache cache = (ClassCache)
48                  ScriptableObject.getTopScopeValue(scope, AKEY);
49          if (cache == null) {
50              throw new RuntimeException("Can't find top level scope for " +
51                      "ClassCache.get");
52          }
53          return cache;
54      }
55  
56      /**
57       * Associate ClassCache object with the given top-level scope.
58       * The ClassCache object can only be associated with the given scope once.
59       *
60       * @param topScope scope to associate this ClassCache object with.
61       * @return true if no previous ClassCache objects were embedded into
62       *         the scope and this ClassCache were successfully associated
63       *         or false otherwise.
64       *
65       * @see #get(Scriptable scope)
66       */
67      public boolean associate(ScriptableObject topScope)
68      {
69          if (topScope.getParentScope() != null) {
70              // Can only associate cache with top level scope
71              throw new IllegalArgumentException();
72          }
73          if (this == topScope.associateValue(AKEY, this)) {
74              associatedScope = topScope;
75              return true;
76          }
77          return false;
78      }

80      /**
81       * Empty caches of generated Java classes and Java reflection information.
82       */
83      public synchronized void clearCaches()
84      {
85          classTable = null;
86          classAdapterCache = null;
87          interfaceAdapterCache = null;
88      }

90      /**
91       * Check if generated Java classes and Java reflection information
92       * is cached.

94      public final boolean isCachingEnabled()
95      {
96          return cachingIsEnabled;

99       /**
100      * Set whether to cache some values.
101      * <p>
102      * By default, the engine will cache the results of
103      * <tt>Class.getMethods()</tt> and similar calls.
104      * This can speed execution dramatically, but increases the memory
105      * footprint. Also, with caching enabled, references may be held to
106      * objects past the lifetime of any real usage.
107      * <p>
108      * If caching is enabled and this method is called with a
109      * <code>false</code> argument, the caches will be emptied.
110      * <p>
111      * Caching is enabled by default.
112      *
113      * @param enabled if true, caching is enabled
114      *
115      * @see #clearCaches()
116      */
117     public synchronized void setCachingEnabled(boolean enabled)
118     {
119         if (enabled == cachingIsEnabled)
120             return;
121         if (!enabled)
122             clearCaches();
123         cachingIsEnabled = enabled;
124     }

176     Object getInterfaceAdapter(Class<?> cl)
177     {
178         return interfaceAdapterCache == null
179                     ? null
180                     : interfaceAdapterCache.get(cl);
181     }

183     synchronized void cacheInterfaceAdapter(Class<?> cl, Object iadapter)
184     {
185         if (cachingIsEnabled) {
186             if (interfaceAdapterCache == null) {
187                 interfaceAdapterCache = new HashMap<Class<?>,Object>();
188             }
189             interfaceAdapterCache.put(cl, iadapter);
190         }
191     }

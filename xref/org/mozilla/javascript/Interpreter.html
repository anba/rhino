116         ContinuationJump(NativeContinuation c, CallFrame current)
117         {
118             this.capturedFrame = (CallFrame)c.getImplementation();
119             if (this.capturedFrame == null || current == null) {
120                 // Continuation and current execution does not share
121                 // any frames if there is nothing to capture or
122                 // if there is no currently executed frames
123                 this.branchFrame = null;
124             } else {
125                 // Search for branch frame where parent frame chains starting
126                 // from captured and current meet.
127                 CallFrame chain1 = this.capturedFrame;
128                 CallFrame chain2 = current;
129 
130                 // First work parents of chain1 or chain2 until the same
131                 // frame depth.
132                 int diff = chain1.frameIndex - chain2.frameIndex;
133                 if (diff != 0) {
134                     if (diff < 0) {
135                         // swap to make sure that
136                         // chain1.frameIndex > chain2.frameIndex and diff > 0
137                         chain1 = current;
138                         chain2 = this.capturedFrame;
139                         diff = -diff;
140                     }
141                     do {
142                         chain1 = chain1.parentFrame;
143                     } while (--diff != 0);
144                     if (chain1.frameIndex != chain2.frameIndex) Kit.codeBug();
145                 }
146 
147                 // Now walk parents in parallel until a shared frame is found
148                 // or until the root is reached.
149                 while (chain1 != chain2 && chain1 != null) {
150                     chain1 = chain1.parentFrame;
151                     chain2 = chain2.parentFrame;
152                 }
153 
154                 this.branchFrame = chain1;
155                 if (this.branchFrame != null && !this.branchFrame.frozen)
156                     Kit.codeBug();
157             }
158         }
159     }

237     private static int getExceptionHandler(CallFrame frame,
238                                            boolean onlyFinally)
239     {
240         int[] exceptionTable = frame.idata.itsExceptionTable;
241         if (exceptionTable == null) {
242             // No exception handlers
243             return -1;
244         }
245 
246         // Icode switch in the interpreter increments PC immediately
247         // and it is necessary to subtract 1 from the saved PC
248         // to point it before the start of the next instruction.
249         int pc = frame.pc - 1;
250 
251         // OPT: use binary search
252         int best = -1, bestStart = 0, bestEnd = 0;
253         for (int i = 0; i != exceptionTable.length; i += EXCEPTION_SLOT_SIZE) {
254             int start = exceptionTable[i + EXCEPTION_TRY_START_SLOT];
255             int end = exceptionTable[i + EXCEPTION_TRY_END_SLOT];
256             if (!(start <= pc && pc < end)) {
257                 continue;
258             }
259             if (onlyFinally && exceptionTable[i + EXCEPTION_TYPE_SLOT] != 1) {
260                 continue;
261             }
262             if (best >= 0) {
263                 // Since handlers always nest and they never have shared end
264                 // although they can share start  it is sufficient to compare
265                 // handlers ends
266                 if (bestEnd < end) {
267                     continue;
268                 }
269                 // Check the above assumption
270                 if (bestStart > start) Kit.codeBug(); // should be nested
271                 if (bestEnd == end) Kit.codeBug();  // no ens sharing
272             }
273             best = i;
274             bestStart = start;
275             bestEnd = end;
276         }
277         return best;
278     }

491     private static int bytecodeSpan(int bytecode)
492     {
493         switch (bytecode) {
494             case Token.THROW :
495             case Token.YIELD:
496             case Icode_GENERATOR:
497             case Icode_GENERATOR_END:
498                 // source line
499                 return 1 + 2;
500 
501             case Icode_GOSUB :
502             case Token.GOTO :
503             case Token.IFEQ :
504             case Token.IFNE :
505             case Icode_IFEQ_POP :
506             case Icode_LEAVEDQ :
507                 // target pc offset
508                 return 1 + 2;
509 
510             case Icode_CALLSPECIAL :
511                 // call type
512                 // is new
513                 // line number
514                 return 1 + 1 + 1 + 2;
515 
516             case Token.CATCH_SCOPE:
517                 // scope flag
518                 return 1 + 1;
519 
520             case Icode_VAR_INC_DEC:
521             case Icode_NAME_INC_DEC:
522             case Icode_PROP_INC_DEC:
523             case Icode_ELEM_INC_DEC:
524             case Icode_REF_INC_DEC:
525                 // type of ++/--
526                 return 1 + 1;
527 
528             case Icode_SHORTNUMBER :
529                 // short number
530                 return 1 + 2;
531 
532             case Icode_INTNUMBER :
533                 // int number
534                 return 1 + 4;
535 
536             case Icode_REG_IND1:
537                 // ubyte index
538                 return 1 + 1;
539 
540             case Icode_REG_IND2:
541                 // ushort index
542                 return 1 + 2;
543 
544             case Icode_REG_IND4:
545                 // int index
546                 return 1 + 4;
547 
548             case Icode_REG_STR1:
549                 // ubyte string index
550                 return 1 + 1;
551 
552             case Icode_REG_STR2:
553                 // ushort string index
554                 return 1 + 2;
555 
556             case Icode_REG_STR4:
557                 // int string index
558                 return 1 + 4;
559 
560             case Icode_GETVAR1:
561             case Icode_SETVAR1:
562             case Icode_SETCONSTVAR1:
563                 // byte var index
564                 return 1 + 1;
565 
566             case Icode_LINE :
567                 // line number
568                 return 1 + 2;
569         }
570         if (!validBytecode(bytecode)) throw Kit.codeBug();
571         return 1;
572     }

575     {
576         UintMap presentLines = new UintMap();
577 
578         byte[] iCode = data.itsICode;
579         int iCodeLength = iCode.length;

581             int bytecode = iCode[pc];
582             int span = bytecodeSpan(bytecode);
583             if (bytecode == Icode_LINE) {

588             pc += span;

594     public void captureStackInfo(RhinoException ex)
595     {
596         Context cx = Context.getCurrentContext();
597         if (cx == null || cx.lastInterpreterFrame == null) {
598             // No interpreter invocations
599             ex.interpreterStackInfo = null;
600             ex.interpreterLineData = null;
601             return;
602         }
603         // has interpreter frame on the stack
604         CallFrame[] array;
605         if (cx.previousInterpreterInvocations == null
606             || cx.previousInterpreterInvocations.size() == 0)
607         {
608             array = new CallFrame[1];
609         } else {
610             int previousCount = cx.previousInterpreterInvocations.size();
611             if (cx.previousInterpreterInvocations.peek()
612                 == cx.lastInterpreterFrame)
613             {
614                 // It can happen if exception was generated after
615                 // frame was pushed to cx.previousInterpreterInvocations
616                 // but before assignment to cx.lastInterpreterFrame.
617                 // In this case frames has to be ignored.
618                 --previousCount;
619             }
620             array = new CallFrame[previousCount + 1];
621             cx.previousInterpreterInvocations.toArray(array);
622         }
623         array[array.length - 1]  = (CallFrame)cx.lastInterpreterFrame;
624 
625         int interpreterFrameCount = 0;
626         for (int i = 0; i != array.length; ++i) {
627             interpreterFrameCount += 1 + array[i].frameIndex;
628         }
629 
630         int[] linePC = new int[interpreterFrameCount];
631         // Fill linePC with pc positions from all interpreter frames.
632         // Start from the most nested frame
633         int linePCIndex = interpreterFrameCount;
634         for (int i = array.length; i != 0;) {
635             --i;
636             CallFrame frame = array[i];
637             while (frame != null) {
638                 --linePCIndex;
639                 linePC[linePCIndex] = frame.pcSourceLineStart;
640                 frame = frame.parentFrame;
641             }
642         }
643         if (linePCIndex != 0) Kit.codeBug();
644 
645         ex.interpreterStackInfo = array;
646         ex.interpreterLineData = linePC;
647     }

661     public String getPatchedStack(RhinoException ex,
662                                   String nativeStackTrace)
663     {
664         String tag = "org.mozilla.javascript.Interpreter.interpretLoop";
665         StringBuffer sb = new StringBuffer(nativeStackTrace.length() + 1000);
666         String lineSeparator = SecurityUtilities.getSystemProperty("line.separator");
667 
668         CallFrame[] array = (CallFrame[])ex.interpreterStackInfo;
669         int[] linePC = ex.interpreterLineData;
670         int arrayIndex = array.length;
671         int linePCIndex = linePC.length;
672         int offset = 0;
673         while (arrayIndex != 0) {
674             --arrayIndex;
675             int pos = nativeStackTrace.indexOf(tag, offset);
676             if (pos < 0) {
677                 break;
678             }
679 
680             // Skip tag length
681             pos += tag.length();
682             // Skip until the end of line
683             for (; pos != nativeStackTrace.length(); ++pos) {
684                 char c = nativeStackTrace.charAt(pos);
685                 if (c == '\n' || c == '\r') {
686                     break;
687                 }
688             }
689             sb.append(nativeStackTrace.substring(offset, pos));
690             offset = pos;
691 
692             CallFrame frame = array[arrayIndex];
693             while (frame != null) {
694                 if (linePCIndex == 0) Kit.codeBug();
695                 --linePCIndex;
696                 InterpreterData idata = frame.idata;
697                 sb.append(lineSeparator);
698                 sb.append("\tat script");
699                 if (idata.itsName != null && idata.itsName.length() != 0) {
700                     sb.append('.');
701                     sb.append(idata.itsName);
702                 }
703                 sb.append('(');
704                 sb.append(idata.itsSourceFile);
705                 int pc = linePC[linePCIndex];
706                 if (pc >= 0) {
707                     // Include line info only if available
708                     sb.append(':');
709                     sb.append(getIndex(idata.itsICode, pc));
710                 }
711                 sb.append(')');
712                 frame = frame.parentFrame;
713             }
714         }
715         sb.append(nativeStackTrace.substring(offset));
716 
717         return sb.toString();
718     }

742         List<ScriptStackElement[]> list = new ArrayList<ScriptStackElement[]>();
743 
744         CallFrame[] array = (CallFrame[])ex.interpreterStackInfo;
745         int[] linePC = ex.interpreterLineData;
746         int arrayIndex = array.length;
747         int linePCIndex = linePC.length;
748         while (arrayIndex != 0) {
749             --arrayIndex;
750             CallFrame frame = array[arrayIndex];
751             List<ScriptStackElement> group = new ArrayList<ScriptStackElement>();
752             while (frame != null) {
753                 if (linePCIndex == 0) Kit.codeBug();
754                 --linePCIndex;
755                 InterpreterData idata = frame.idata;
756                 String fileName = idata.itsSourceFile;
757                 String functionName = null;
758                 int lineNumber = -1;
759                 int pc = linePC[linePCIndex];
760                 if (pc >= 0) {
761                     lineNumber = getIndex(idata.itsICode, pc);
762                 }
763                 if (idata.itsName != null && idata.itsName.length() != 0) {
764                     functionName = idata.itsName;
765                 }
766                 frame = frame.parentFrame;
767                 group.add(new ScriptStackElement(fileName, functionName, lineNumber));
768             }
769             list.add(group.toArray(new ScriptStackElement[group.size()]));
770         }
771         return list.toArray(new ScriptStackElement[list.size()][]);
772     }

828     public static Object resumeGenerator(Context cx,
829                                          Scriptable scope,
830                                          int operation,
831                                          Object savedState,
832                                          Object value)
833     {
834       CallFrame frame = (CallFrame) savedState;
835       GeneratorState generatorState = new GeneratorState(operation, value);
836       if (operation == NativeGenerator.GENERATOR_CLOSE) {
837           try {
838               return interpretLoop(cx, frame, generatorState);
839           } catch (RuntimeException e) {
840               // Only propagate exceptions other than closingException
841               if (e != value)
842                   throw e;
843           }
844           return Undefined.instance;
845       }
846       Object result = interpretLoop(cx, frame, generatorState);
847       if (generatorState.returnedException != null)
848           throw generatorState.returnedException;
849       return result;
850     }

853                                              Scriptable scope, Object[] args)
854     {
855         if (!ScriptRuntime.hasTopCall(cx)) {

873

878     private static Object interpretLoop(Context cx, CallFrame frame,
879                                         Object throwable)
880     {
881         // throwable holds exception object to rethrow or catch
882         // It is also used for continuation restart in which case
883         // it holds ContinuationJump
884 
885         final Object DBL_MRK = DOUBLE_MARK;
886         final Object undefined = Undefined.instance;
887 
888         final boolean instructionCounting = (cx.instructionThreshold != 0);
889         // arbitrary number to add to instructionCount when calling
890         // other functions
891         final int INVOCATION_COST = 100;
892         // arbitrary exception cost for instruction counting
893         final int EXCEPTION_COST = 100;
894 
895         String stringReg = null;
896         int indexReg = -1;
897 
898         if (cx.lastInterpreterFrame != null) {
899             // save the top frame from the previous interpretLoop
900             // invocation on the stack
901             if (cx.previousInterpreterInvocations == null) {
902                 cx.previousInterpreterInvocations = new ObjArray();
903             }
904             cx.previousInterpreterInvocations.push(cx.lastInterpreterFrame);
905         }
906 
907         // When restarting continuation throwable is not null and to jump
908         // to the code that rewind continuation state indexReg should be set
909         // to -1.
910         // With the normal call throwable == null and indexReg == -1 allows to
911         // catch bugs with using indeReg to access array elements before
912         // initializing indexReg.
913 
914         GeneratorState generatorState = null;
915         if (throwable != null) {
916             if (throwable instanceof GeneratorState) {
917               generatorState = (GeneratorState) throwable;
918 
919               // reestablish this call frame
920               enterFrame(cx, frame, ScriptRuntime.emptyArgs, true);
921               throwable = null;
922             } else if (!(throwable instanceof ContinuationJump)) {
923                 // It should be continuation
924                 Kit.codeBug();
925             }
926         }
927 
928         Object interpreterResult = null;
929         double interpreterResultDbl = 0.0;
930 
931         StateLoop: for (;;) {
932             withoutExceptions: try {
933 
934                 if (throwable != null) {
935                     // Need to return both 'frame' and 'throwable' from
936                     // 'processThrowable', so just added a 'throwable'
937                     // member in 'frame'.
938                     frame = processThrowable(cx, throwable, frame, indexReg,
939                                              instructionCounting);
940                     throwable = frame.throwable;
941                     frame.throwable = null;
942                 } else {
943                     if (generatorState == null && frame.frozen) Kit.codeBug();
944                 }
945 
946                 // Use local variables for constant values in frame
947                 // for faster access
948                 Object[] stack = frame.stack;
949                 double[] sDbl = frame.sDbl;
950                 Object[] vars = frame.varSource.stack;
951                 double[] varDbls = frame.varSource.sDbl;
952                 int[] varAttributes = frame.varSource.stackAttributes;
953                 byte[] iCode = frame.idata.itsICode;
954                 String[] strings = frame.idata.itsStringTable;
955 
956                 // Use local for stackTop as well. Since execption handlers
957                 // can only exist at statement level where stack is empty,
958                 // it is necessary to save/restore stackTop only across
959                 // function calls and normal returns.
960                 int stackTop = frame.savedStackTop;
961 
962                 // Store new frame in cx which is used for error reporting etc.
963                 cx.lastInterpreterFrame = frame;
964 
965                 Loop: for (;;) {
966 
967                     // Exception handler assumes that PC is already incremented
968                     // pass the instruction start when it searches the
969                     // exception handler
970                     int op = iCode[frame.pc++];
971                     jumplessRun: {
972 
973     // Back indent to ease implementation reading
974 switch (op) {
975     case Icode_GENERATOR: {
976         if (!frame.frozen) {
977           // First time encountering this opcode: create new generator
978           // object and return
979           frame.pc--; // we want to come back here when we resume
980           CallFrame generatorFrame = captureFrameForGenerator(frame);
981           generatorFrame.frozen = true;
982           NativeGenerator generator = new NativeGenerator(frame.scope,
983               generatorFrame.fnOrScript, generatorFrame);
984           frame.result = generator;
985           break Loop;
986         } else {
987           // We are now resuming execution. Fall through to YIELD case.
988         }
989     }
990     // fall through...
991     case Token.YIELD: {
992         if (!frame.frozen) {
993             return freezeGenerator(cx, frame, stackTop, generatorState);
994         } else {
995             Object obj = thawGenerator(frame, stackTop, generatorState, op);
996             if (obj != Scriptable.NOT_FOUND) {
997                 throwable = obj;
998                 break withoutExceptions;
999             }
1000             continue Loop;
1001         }
1002     }
1003     case Icode_GENERATOR_END: {
1004       // throw StopIteration
1005       frame.frozen = true;
1006       int sourceLine = getIndex(iCode, frame.pc);
1007       generatorState.returnedException = new JavaScriptException(
1008           NativeIterator.getStopIterationObject(frame.scope),
1009           frame.idata.itsSourceFile, sourceLine);
1010       break Loop;
1011     }
1012     case Token.THROW: {
1013         Object value = stack[stackTop];
1014         if (value == DBL_MRK) value = ScriptRuntime.wrapNumber(sDbl[stackTop]);
1015         --stackTop;
1016 
1017         int sourceLine = getIndex(iCode, frame.pc);
1018         throwable = new JavaScriptException(value,
1019                                             frame.idata.itsSourceFile,
1020                                             sourceLine);
1021         break withoutExceptions;
1022     }
1023     case Token.RETHROW: {
1024         indexReg += frame.localShift;
1025         throwable = stack[indexReg];
1026         break withoutExceptions;
1027     }
1028     case Token.GE :
1029     case Token.LE :
1030     case Token.GT :
1031     case Token.LT : {
1032         stackTop = doCompare(frame, op, stack, sDbl, stackTop);
1033         continue Loop;
1034     }
1035     case Token.IN :
1036     case Token.INSTANCEOF : {
1037         stackTop = doInOrInstanceof(cx, op, stack, sDbl, stackTop);
1038         continue Loop;
1039     }
1040     case Token.EQ :
1041     case Token.NE : {
1042         --stackTop;
1043         boolean valBln = doEquals(stack, sDbl, stackTop);
1044         valBln ^= (op == Token.NE);
1045         stack[stackTop] = ScriptRuntime.wrapBoolean(valBln);
1046         continue Loop;
1047     }
1048     case Token.SHEQ :
1049     case Token.SHNE : {
1050         --stackTop;
1051         boolean valBln = doShallowEquals(stack, sDbl, stackTop);
1052         valBln ^= (op == Token.SHNE);
1053         stack[stackTop] = ScriptRuntime.wrapBoolean(valBln);
1054         continue Loop;
1055     }
1056     case Token.IFNE :
1057         if (stack_boolean(frame, stackTop--)) {
1058             frame.pc += 2;
1059             continue Loop;
1060         }
1061         break jumplessRun;
1062     case Token.IFEQ :
1063         if (!stack_boolean(frame, stackTop--)) {
1064             frame.pc += 2;
1065             continue Loop;
1066         }
1067         break jumplessRun;
1068     case Icode_IFEQ_POP :
1069         if (!stack_boolean(frame, stackTop--)) {
1070             frame.pc += 2;
1071             continue Loop;
1072         }
1073         stack[stackTop--] = null;
1074         break jumplessRun;
1075     case Token.GOTO :
1076         break jumplessRun;
1077     case Icode_GOSUB :
1078         ++stackTop;
1079         stack[stackTop] = DBL_MRK;
1080         sDbl[stackTop] = frame.pc + 2;
1081         break jumplessRun;
1082     case Icode_STARTSUB :
1083         if (stackTop == frame.emptyStackTop + 1) {
1084             // Call from Icode_GOSUB: store return PC address in the local
1085             indexReg += frame.localShift;
1086             stack[indexReg] = stack[stackTop];
1087             sDbl[indexReg] = sDbl[stackTop];
1088             --stackTop;
1089         } else {
1090             // Call from exception handler: exception object is already stored
1091             // in the local
1092             if (stackTop != frame.emptyStackTop) Kit.codeBug();
1093         }
1094         continue Loop;
1095     case Icode_RETSUB : {
1096         // indexReg: local to store return address
1097         if (instructionCounting) {
1098             addInstructionCount(cx, frame, 0);
1099         }
1100         indexReg += frame.localShift;
1101         Object value = stack[indexReg];
1102         if (value != DBL_MRK) {
1103             // Invocation from exception handler, restore object to rethrow
1104             throwable = value;
1105             break withoutExceptions;
1106         }
1107         // Normal return from GOSUB
1108         frame.pc = (int)sDbl[indexReg];
1109         if (instructionCounting) {
1110             frame.pcPrevBranch = frame.pc;
1111         }
1112         continue Loop;
1113     }
1114     case Icode_POP :
1115         stack[stackTop] = null;
1116         stackTop--;
1117         continue Loop;
1118     case Icode_POP_RESULT :
1119         frame.result = stack[stackTop];
1120         frame.resultDbl = sDbl[stackTop];
1121         stack[stackTop] = null;
1122         --stackTop;
1123         continue Loop;
1124     case Icode_DUP :
1125         stack[stackTop + 1] = stack[stackTop];
1126         sDbl[stackTop + 1] = sDbl[stackTop];
1127         stackTop++;
1128         continue Loop;
1129     case Icode_DUP2 :
1130         stack[stackTop + 1] = stack[stackTop - 1];
1131         sDbl[stackTop + 1] = sDbl[stackTop - 1];
1132         stack[stackTop + 2] = stack[stackTop];
1133         sDbl[stackTop + 2] = sDbl[stackTop];
1134         stackTop += 2;
1135         continue Loop;
1136     case Icode_SWAP : {
1137         Object o = stack[stackTop];
1138         stack[stackTop] = stack[stackTop - 1];
1139         stack[stackTop - 1] = o;
1140         double d = sDbl[stackTop];
1141         sDbl[stackTop] = sDbl[stackTop - 1];
1142         sDbl[stackTop - 1] = d;
1143         continue Loop;
1144     }
1145     case Token.RETURN :
1146         frame.result = stack[stackTop];
1147         frame.resultDbl = sDbl[stackTop];
1148         --stackTop;
1149         break Loop;
1150     case Token.RETURN_RESULT :
1151         break Loop;
1152     case Icode_RETUNDEF :
1153         frame.result = undefined;
1154         break Loop;
1155     case Token.BITNOT : {
1156         int rIntValue = stack_int32(frame, stackTop);
1157         stack[stackTop] = DBL_MRK;
1158         sDbl[stackTop] = ~rIntValue;
1159         continue Loop;
1160     }
1161     case Token.BITAND :
1162     case Token.BITOR :
1163     case Token.BITXOR :
1164     case Token.LSH :
1165     case Token.RSH : {
1166         stackTop = doBitOp(frame, op, stack, sDbl, stackTop);
1167         continue Loop;
1168     }
1169     case Token.URSH : {
1170         double lDbl = stack_double(frame, stackTop - 1);
1171         int rIntValue = stack_int32(frame, stackTop) & 0x1F;
1172         stack[--stackTop] = DBL_MRK;
1173         sDbl[stackTop] = ScriptRuntime.toUint32(lDbl) >>> rIntValue;
1174         continue Loop;
1175     }
1176     case Token.NEG :
1177     case Token.POS : {
1178         double rDbl = stack_double(frame, stackTop);
1179         stack[stackTop] = DBL_MRK;
1180         if (op == Token.NEG) {
1181             rDbl = -rDbl;
1182         }
1183         sDbl[stackTop] = rDbl;
1184         continue Loop;
1185     }
1186     case Token.ADD :
1187         --stackTop;
1188         doAdd(stack, sDbl, stackTop, cx);
1189         continue Loop;
1190     case Token.SUB :
1191     case Token.MUL :
1192     case Token.DIV :
1193     case Token.MOD : {
1194         stackTop = doArithmetic(frame, op, stack, sDbl, stackTop);
1195         continue Loop;
1196     }
1197     case Token.NOT :
1198         stack[stackTop] = ScriptRuntime.wrapBoolean(
1199                               !stack_boolean(frame, stackTop));
1200         continue Loop;
1201     case Token.BINDNAME :
1202         stack[++stackTop] = ScriptRuntime.bind(cx, frame.scope, stringReg);
1203         continue Loop;
1204     case Token.STRICT_SETNAME:
1205     case Token.SETNAME : {
1206         Object rhs = stack[stackTop];
1207         if (rhs == DBL_MRK) rhs = ScriptRuntime.wrapNumber(sDbl[stackTop]);
1208         --stackTop;
1209         Scriptable lhs = (Scriptable)stack[stackTop];
1210         stack[stackTop] = op == Token.SETNAME ?
1211                 ScriptRuntime.setName(lhs, rhs, cx,
1212                                       frame.scope, stringReg) :
1213                 ScriptRuntime.strictSetName(lhs, rhs, cx,
1214                                       frame.scope, stringReg);
1215         continue Loop;
1216     }
1217     case Icode_SETCONST: {
1218         Object rhs = stack[stackTop];
1219         if (rhs == DBL_MRK) rhs = ScriptRuntime.wrapNumber(sDbl[stackTop]);
1220         --stackTop;
1221         Scriptable lhs = (Scriptable)stack[stackTop];
1222         stack[stackTop] = ScriptRuntime.setConst(lhs, rhs, cx, stringReg);
1223         continue Loop;
1224     }
1225     case Token.DELPROP :
1226     case Icode_DELNAME : {
1227         stackTop = doDelName(cx, op, stack, sDbl, stackTop);
1228         continue Loop;
1229     }
1230     case Token.GETPROPNOWARN : {
1231         Object lhs = stack[stackTop];
1232         if (lhs == DBL_MRK) lhs = ScriptRuntime.wrapNumber(sDbl[stackTop]);
1233         stack[stackTop] = ScriptRuntime.getObjectPropNoWarn(lhs, stringReg, cx);
1234         continue Loop;
1235     }
1236     case Token.GETPROP : {
1237         Object lhs = stack[stackTop];
1238         if (lhs == DBL_MRK) lhs = ScriptRuntime.wrapNumber(sDbl[stackTop]);
1239         stack[stackTop] = ScriptRuntime.getObjectProp(lhs, stringReg, cx, frame.scope);
1240         continue Loop;
1241     }
1242     case Token.SETPROP : {
1243         Object rhs = stack[stackTop];
1244         if (rhs == DBL_MRK) rhs = ScriptRuntime.wrapNumber(sDbl[stackTop]);
1245         --stackTop;
1246         Object lhs = stack[stackTop];
1247         if (lhs == DBL_MRK) lhs = ScriptRuntime.wrapNumber(sDbl[stackTop]);
1248         stack[stackTop] = ScriptRuntime.setObjectProp(lhs, stringReg, rhs, cx);
1249         continue Loop;
1250     }
1251     case Icode_PROP_INC_DEC : {
1252         Object lhs = stack[stackTop];
1253         if (lhs == DBL_MRK) lhs = ScriptRuntime.wrapNumber(sDbl[stackTop]);
1254         stack[stackTop] = ScriptRuntime.propIncrDecr(lhs, stringReg,
1255                                                      cx, iCode[frame.pc]);
1256         ++frame.pc;
1257         continue Loop;
1258     }
1259     case Token.GETELEM : {
1260         stackTop = doGetElem(cx, frame, stack, sDbl, stackTop);
1261         continue Loop;
1262     }
1263     case Token.SETELEM : {
1264         stackTop = doSetElem(cx, stack, sDbl, stackTop);
1265         continue Loop;
1266     }
1267     case Icode_ELEM_INC_DEC: {
1268         stackTop = doElemIncDec(cx, frame, iCode, stack, sDbl, stackTop);
1269         continue Loop;
1270     }
1271     case Token.GET_REF : {
1272         Ref ref = (Ref)stack[stackTop];
1273         stack[stackTop] = ScriptRuntime.refGet(ref, cx);
1274         continue Loop;
1275     }
1276     case Token.SET_REF : {
1277         Object value = stack[stackTop];
1278         if (value == DBL_MRK) value = ScriptRuntime.wrapNumber(sDbl[stackTop]);
1279         --stackTop;
1280         Ref ref = (Ref)stack[stackTop];
1281         stack[stackTop] = ScriptRuntime.refSet(ref, value, cx);
1282         continue Loop;
1283     }
1284     case Token.DEL_REF : {
1285         Ref ref = (Ref)stack[stackTop];
1286         stack[stackTop] = ScriptRuntime.refDel(ref, cx);
1287         continue Loop;
1288     }
1289     case Icode_REF_INC_DEC : {
1290         Ref ref = (Ref)stack[stackTop];
1291         stack[stackTop] = ScriptRuntime.refIncrDecr(ref, cx, iCode[frame.pc]);
1292         ++frame.pc;
1293         continue Loop;
1294     }
1295     case Token.LOCAL_LOAD :
1296         ++stackTop;
1297         indexReg += frame.localShift;
1298         stack[stackTop] = stack[indexReg];
1299         sDbl[stackTop] = sDbl[indexReg];
1300         continue Loop;
1301     case Icode_LOCAL_CLEAR :
1302         indexReg += frame.localShift;
1303         stack[indexReg] = null;
1304         continue Loop;
1305     case Icode_NAME_AND_THIS :
1306         // stringReg: name
1307         ++stackTop;
1308         stack[stackTop] = ScriptRuntime.getNameFunctionAndThis(stringReg,
1309                                                                cx, frame.scope);
1310         ++stackTop;
1311         stack[stackTop] = ScriptRuntime.lastStoredScriptable(cx);
1312         continue Loop;
1313     case Icode_PROP_AND_THIS: {
1314         Object obj = stack[stackTop];
1315         if (obj == DBL_MRK) obj = ScriptRuntime.wrapNumber(sDbl[stackTop]);
1316         // stringReg: property
1317         stack[stackTop] = ScriptRuntime.getPropFunctionAndThis(obj, stringReg,
1318                                                                cx, frame.scope);
1319         ++stackTop;
1320         stack[stackTop] = ScriptRuntime.lastStoredScriptable(cx);
1321         continue Loop;
1322     }
1323     case Icode_ELEM_AND_THIS: {
1324         Object obj = stack[stackTop - 1];
1325         if (obj == DBL_MRK) obj = ScriptRuntime.wrapNumber(sDbl[stackTop - 1]);
1326         Object id = stack[stackTop];
1327         if (id == DBL_MRK) id = ScriptRuntime.wrapNumber(sDbl[stackTop]);
1328         stack[stackTop - 1] = ScriptRuntime.getElemFunctionAndThis(obj, id, cx);
1329         stack[stackTop] = ScriptRuntime.lastStoredScriptable(cx);
1330         continue Loop;
1331     }
1332     case Icode_VALUE_AND_THIS : {
1333         Object value = stack[stackTop];
1334         if (value == DBL_MRK) value = ScriptRuntime.wrapNumber(sDbl[stackTop]);
1335         stack[stackTop] = ScriptRuntime.getValueFunctionAndThis(value, cx);
1336         ++stackTop;
1337         stack[stackTop] = ScriptRuntime.lastStoredScriptable(cx);
1338         continue Loop;
1339     }
1340     case Icode_CALLSPECIAL : {
1341         if (instructionCounting) {
1342             cx.instructionCount += INVOCATION_COST;
1343         }
1344         stackTop = doCallSpecial(cx, frame, stack, sDbl, stackTop, iCode, indexReg);
1345         continue Loop;
1346     }
1347     case Token.CALL :
1348     case Icode_TAIL_CALL :
1349     case Token.REF_CALL : {
1350         if (instructionCounting) {
1351             cx.instructionCount += INVOCATION_COST;
1352         }
1353         // stack change: function thisObj arg0 .. argN -> result
1354         // indexReg: number of arguments
1355         stackTop -= 1 + indexReg;
1356 
1357         // CALL generation ensures that fun and funThisObj
1358         // are already Scriptable and Callable objects respectively
1359         Callable fun = (Callable)stack[stackTop];
1360         Scriptable funThisObj = (Scriptable)stack[stackTop + 1];
1361         if (op == Token.REF_CALL) {
1362             Object[] outArgs = getArgsArray(stack, sDbl, stackTop + 2,
1363                                             indexReg);
1364             stack[stackTop] = ScriptRuntime.callRef(fun, funThisObj,
1365                                                     outArgs, cx);
1366             continue Loop;
1367         }
1368         Scriptable calleeScope = frame.scope;
1369         if (frame.useActivation) {
1370             calleeScope = ScriptableObject.getTopLevelScope(frame.scope);
1371         }
1372         if (fun instanceof InterpretedFunction) {
1373             InterpretedFunction ifun = (InterpretedFunction)fun;
1374             if (frame.fnOrScript.securityDomain == ifun.securityDomain) {
1375                 CallFrame callParentFrame = frame;
1376                 CallFrame calleeFrame = new CallFrame();

2041                     // Restarting detached continuation
2042                     indexReg = -1;
2043                     continue StateLoop;

2045                 // Return continuation result to the caller
2046                 interpreterResult = cjump.result;
2047                 interpreterResultDbl = cjump.resultDbl;
2048                 throwable = null;
2049             }
2050             break StateLoop;
2051 
2052         } // end of StateLoop: for(;;)
2053 
2054         // Do cleanups/restorations before the final return or throw
2055

2057             && cx.previousInterpreterInvocations.size() != 0)
2058         {
2059             cx.lastInterpreterFrame
2060                 = cx.previousInterpreterInvocations.pop();
2061         } else {
2062             // It was the last interpreter frame on the stack
2063             cx.lastInterpreterFrame = null;
2064             // Force GC of the value cx.previousInterpreterInvocations
2065             cx.previousInterpreterInvocations = null;
2066         }
2067 
2068         if (throwable != null) {
2069             if (throwable instanceof RuntimeException) {
2070                 throw (RuntimeException)throwable;
2071             } else {
2072                 // Must be instance of Error or code bug
2073                 throw (Error)throwable;
2074             }
2075         }
2076

2079                : ScriptRuntime.wrapNumber(interpreterResultDbl);

2513         // the index of handler
2514

2517             // control appropriately

2557             // and all frames in between
2558             int rewindCount = cjump.capturedFrame.frameIndex + 1;
2559             if (cjump.branchFrame != null) {
2560                 rewindCount -= cjump.branchFrame.frameIndex;
2561             }
2562

2565

2568                 if (!x.frozen) Kit.codeBug();

2574                         enterFrames = new CallFrame[rewindCount
2575                                                     - i];
2576                     }
2577                     enterFrames[enterCount] = x;
2578                     ++enterCount;
2579                 }
2580                 x = x.parentFrame;
2581             }
2582

2585                 // order since they were stored starting from
2586                 // the capturedFrame, not branchFrame

2590             }

2594             // continuation, so clone capturedFrame and
2595             // emulate return that function with the suplied result
2596             frame = cjump.capturedFrame.cloneFrozen();

7   package org.mozilla.javascript;
8   
9   import java.io.Serializable;
10  
11  /**
12   * <p>This class represents a string composed of two components, each of which
13   * may be a <code>java.lang.String</code> or another ConsString.</p>
14   *
15   * <p>This string representation is optimized for concatenation using the "+"
16   * operator. Instead of immediately copying both components to a new character
17   * array, ConsString keeps references to the original components and only
18   * converts them to a String if either toString() is called or a certain depth
19   * level is reached.</p>
20   *
21   * <p>Note that instances of this class are only immutable if both parts are
22   * immutable, i.e. either Strings or ConsStrings that are ultimately composed
23   * of Strings.</p>
24   *
25   * <p>Both the name and the concept are borrowed from V8.</p>
26   */
27  public class ConsString implements CharSequence, Serializable {
28  
29      private static final long serialVersionUID = -8432806714471372570L;
30  
31      private CharSequence s1, s2;
32      private final int length;
33      private int depth;
34  
35      public ConsString(CharSequence str1, CharSequence str2) {
36          s1 = str1;
37          s2 = str2;
38          length = str1.length() + str2.length();
39          depth = 1;
40          if (str1 instanceof ConsString) {
41              depth += ((ConsString)str1).depth;
42          }
43          if (str2 instanceof ConsString) {
44              depth += ((ConsString)str2).depth;
45          }
46          // Don't let it grow too deep, can cause stack overflows
47          if (depth > 2000) {
48              flatten();
49          }
50      }
51  
52      // Replace with string representation when serializing
53      private Object writeReplace() {
54          return this.toString();
55      }
56      
57      public String toString() {
58          return depth == 0 ? (String)s1 : flatten();
59      }
60  
61      private synchronized String flatten() {
62          if (depth > 0) {
63              StringBuilder b = new StringBuilder(length);
64              appendTo(b);
65              s1 = b.toString();
66              s2 = "";
67              depth = 0;
68          }
69          return (String)s1;
70      }
71  
72      private synchronized void appendTo(StringBuilder b) {
73          appendFragment(s1, b);
74          appendFragment(s2, b);
75      }
76  
77      private static void appendFragment(CharSequence s, StringBuilder b) {
78          if (s instanceof ConsString) {
79              ((ConsString)s).appendTo(b);
80          } else {
81              b.append(s);
82          }
83      }
84  
85      public int length() {
86          return length;
87      }
88  
89      public char charAt(int index) {
90          String str = depth == 0 ? (String)s1 : flatten();
91          return str.charAt(index);
92      }
93  
94      public CharSequence subSequence(int start, int end) {
95          String str = depth == 0 ? (String)s1 : flatten();
96          return str.substring(start, end);
97      }
98  
99  }

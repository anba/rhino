5   package org.mozilla.javascript.commonjs.module.provider;
6   
7   import java.io.Serializable;
8   import java.util.StringTokenizer;
9   
10  /**
11   * Breaks a "contentType; charset=encoding" MIME type into content type and
12   * encoding parts.
13   * @author Attila Szegedi
14   * @version $Id: ParsedContentType.java,v 1.3 2011/04/07 20:26:12 hannes%helma.at Exp $
15   */
16  public final class ParsedContentType implements Serializable
17  {
18      private static final long serialVersionUID = 1L;
19  
20      private final String contentType;
21      private final String encoding;
22  
23      /**
24       * Creates a new parsed content type.
25       * @param mimeType the full MIME type; typically the value of the
26       * "Content-Type" header of some MIME-compliant message. Can be null.
27       */
28      public ParsedContentType(String mimeType) {
29          String contentType = null;
30          String encoding = null;
31          if(mimeType != null) {
32              StringTokenizer tok = new StringTokenizer(mimeType, ";");
33              if(tok.hasMoreTokens()) {
34                  contentType = tok.nextToken().trim();
35                  while(tok.hasMoreTokens()) {
36                      String param = tok.nextToken().trim();
37                      if(param.startsWith("charset=")) {
38                          encoding = param.substring(8).trim();
39                          int l = encoding.length();
40                          if(l > 0) {
41                              if(encoding.charAt(0) == '"') {
42                                  encoding = encoding.substring(1);
43                              }
44                              if(encoding.charAt(l - 1) == '"') {
45                                  encoding = encoding.substring(0, l - 1);
46                              }
47                          }
48                          break;
49                      }
50                  }
51              }
52          }
53          this.contentType = contentType;
54          this.encoding = encoding;
55      }
56  
57      /**
58       * Returns the content type (without charset declaration) of the MIME type.
59       * @return the content type (without charset declaration) of the MIME type.
60       * Can be null if the MIME type was null.
61       */
62      public String getContentType() {
63          return contentType;
64      }
65  
66      /**
67       * Returns the character encoding of the MIME type.
68       * @return the character encoding of the MIME type. Can be null when it is
69       * not specified.
70       */
71      public String getEncoding() {
72          return encoding;
73      }
74  }

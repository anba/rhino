23  /**
24   * A module script provider that uses a module source provider to load modules
25   * and caches the loaded modules. It softly references the loaded modules'
26   * Rhino {@link Script} objects, thus a module once loaded can become eligible
27   * for garbage collection if it is otherwise unused under memory pressure.
28   * Instances of this class are thread safe.
29   * @author Attila Szegedi
30   * @version $Id: SoftCachingModuleScriptProvider.java,v 1.3 2011/04/07 20:26:12 hannes%helma.at Exp $
31   */
32  public class SoftCachingModuleScriptProvider extends CachingModuleScriptProviderBase
33  {
34      private static final long serialVersionUID = 1L;
35  
36      private transient ReferenceQueue<Script> scriptRefQueue =
37          new ReferenceQueue<Script>();
38  
39      private transient ConcurrentMap<String, ScriptReference> scripts =
40          new ConcurrentHashMap<String, ScriptReference>(16, .75f,
41                  getConcurrencyLevel());
42  
43      /**
44       * Creates a new module provider with the specified module source provider.
45       * @param moduleSourceProvider provider for modules' source code
46       */
47      public SoftCachingModuleScriptProvider(
48              ModuleSourceProvider moduleSourceProvider)
49      {
50          super(moduleSourceProvider);
51      }
52  
53      @Override
54      public ModuleScript getModuleScript(Context cx, String moduleId,
55              URI uri, URI base, Scriptable paths)
56              throws Exception
57      {
58          // Overridden to clear the reference queue before retrieving the
59          // script.
60          for(;;) {
61              ScriptReference ref = (ScriptReference)scriptRefQueue.poll();
62              if(ref == null) {
63                  break;
64              }
65              scripts.remove(ref.getModuleId(), ref);
66          }
67          return super.getModuleScript(cx, moduleId, uri, base, paths);
68      }
69  
70      @Override
71      protected CachedModuleScript getLoadedModule(String moduleId) {
72          final ScriptReference scriptRef = scripts.get(moduleId);
73          return scriptRef != null ? scriptRef.getCachedModuleScript() : null;
74      }
75  
76      @Override
77      protected void putLoadedModule(String moduleId, ModuleScript moduleScript,
78              Object validator)
79      {
80          scripts.put(moduleId, new ScriptReference(moduleScript.getScript(),
81                  moduleId, moduleScript.getUri(), moduleScript.getBase(),
82                  validator, scriptRefQueue));
83      }
84  
85      private static class ScriptReference extends SoftReference<Script> {
86          private final String moduleId;
87          private final URI uri;
88          private final URI base;
89          private final Object validator;
90  
91          ScriptReference(Script script, String moduleId, URI uri, URI base,
92                  Object validator, ReferenceQueue<Script> refQueue) {
93              super(script, refQueue);
94              this.moduleId = moduleId;
95              this.uri = uri;
96              this.base = base;
97              this.validator = validator;
98          }
99  
100         CachedModuleScript getCachedModuleScript() {
101             final Script script = get();
102             if(script == null) {
103                 return null;
104             }
105             return new CachedModuleScript(new ModuleScript(script, uri, base),
106                     validator);
107         }
108 
109         String getModuleId() {
110             return moduleId;
111         }
112     }
113 
114     private void readObject(ObjectInputStream in) throws IOException,
115     ClassNotFoundException
116     {
117         scriptRefQueue = new ReferenceQueue<Script>();
118         scripts = new ConcurrentHashMap<String, ScriptReference>();
119         final Map<String, CachedModuleScript> serScripts = (Map)in.readObject();
120         for(Map.Entry<String, CachedModuleScript> entry: serScripts.entrySet()) {
121             final CachedModuleScript cachedModuleScript = entry.getValue();
122             putLoadedModule(entry.getKey(), cachedModuleScript.getModule(),
123                     cachedModuleScript.getValidator());
124         }
125     }
126 
127     private void writeObject(ObjectOutputStream out) throws IOException {
128         final Map<String, CachedModuleScript> serScripts =
129             new HashMap<String, CachedModuleScript>();
130         for(Map.Entry<String, ScriptReference> entry: scripts.entrySet()) {
131             final CachedModuleScript cachedModuleScript =
132                 entry.getValue().getCachedModuleScript();
133             if(cachedModuleScript != null) {
134                 serScripts.put(entry.getKey(), cachedModuleScript);
135             }
136         }
137         out.writeObject(serScripts);
138     }

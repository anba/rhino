21  /**
22   * Implements the require() function as defined by
23   * <a href="http://wiki.commonjs.org/wiki/Modules/1.1">Common JS modules</a>.
24   * <h1>Thread safety</h1>
25   * You will ordinarily create one instance of require() for every top-level
26   * scope. This ordinarily means one instance per program execution, except if
27   * you use shared top-level scopes and installing most objects into them.
28   * Module loading is thread safe, so using a single require() in a shared
29   * top-level scope is also safe.
30   * <h1>Creation</h1>
31   * If you need to create many otherwise identical require() functions for
32   * different scopes, you might want to use {@link RequireBuilder} for
33   * convenience.
34   * <h1>Making it available</h1>
35   * In order to make the require() function available to your JavaScript
36   * program, you need to invoke either {@link #install(Scriptable)} or
37   * {@link #requireMain(Context, String)}.
38   * @author Attila Szegedi
39   * @version $Id: Require.java,v 1.4 2011/04/07 20:26:11 hannes%helma.at Exp $
40   */

63      /**
64       * Creates a new instance of the require() function. Upon constructing it,
65       * you will either want to install it in the global (or some other) scope
66       * using {@link #install(Scriptable)}, or alternatively, you can load the
67       * program's main module using {@link #requireMain(Context, String)} and
68       * then act on the main module's exports.
69       * @param cx the current context
70       * @param nativeScope a scope that provides the standard native JavaScript
71       * objects.
72       * @param moduleScriptProvider a provider for module scripts
73       * @param preExec an optional script that is executed in every module's
74       * scope before its module script is run.
75       * @param postExec an optional script that is executed in every module's
76       * scope after its module script is run.
77       * @param sandboxed if set to true, the require function will be sandboxed.
78       * This means that it doesn't have the "paths" property, and also that the
79       * modules it loads don't export the "module.uri" property.
80       */
81      public Require(Context cx, Scriptable nativeScope,
82              ModuleScriptProvider moduleScriptProvider, Script preExec,
83              Script postExec, boolean sandboxed) {
84          this.moduleScriptProvider = moduleScriptProvider;
85          this.nativeScope = nativeScope;
86          this.sandboxed = sandboxed;
87          this.preExec = preExec;
88          this.postExec = postExec;
89          setPrototype(ScriptableObject.getFunctionPrototype(nativeScope));
90          if(!sandboxed) {
91              paths = cx.newArray(nativeScope, 0);
92              defineReadOnlyProperty(this, "paths", paths);
93          }
94          else {
95              paths = null;
96          }
97      }

99      /**
100      * Calling this method establishes a module as being the main module of the
101      * program to which this require() instance belongs. The module will be
102      * loaded as if require()'d and its "module" property will be set as the
103      * "main" property of this require() instance. You have to call this method
104      * before the module has been loaded (that is, the call to this method must
105      * be the first to require the module and thus trigger its loading). Note
106      * that the main module will execute in its own scope and not in the global
107      * scope. Since all other modules see the global scope, executing the main
108      * module in the global scope would open it for tampering by other modules.
109      * @param cx the current context
110      * @param mainModuleId the ID of the main module
111      * @return the "exports" property of the main module
112      * @throws IllegalStateException if the main module is already loaded when
113      * required, or if this require() instance already has a different main
114      * module set.
115      */
116     public Scriptable requireMain(Context cx, String mainModuleId) {
117         if(this.mainModuleId != null) {
118             if (!this.mainModuleId.equals(mainModuleId)) {
119                 throw new IllegalStateException("Main module already set to " +
120                     this.mainModuleId);
121             }
122             return mainExports;
123         }
124 
125         ModuleScript moduleScript;
126         try {
127             // try to get the module script to see if it is on the module path
128             moduleScript = moduleScriptProvider.getModuleScript(
129                     cx, mainModuleId, null, null, paths);
130         } catch (RuntimeException x) {
131             throw x;
132         } catch (Exception x) {
133             throw new RuntimeException(x);
134         }
135 
136         if (moduleScript != null) {
137             mainExports = getExportedModuleInterface(cx, mainModuleId,
138                     null, null, true);
139         } else if (!sandboxed) {
140 
141             URI mainUri = null;
142 
143             // try to resolve to an absolute URI or file path
144             try {
145                 mainUri = new URI(mainModuleId);
146             } catch (URISyntaxException usx) {
147                 // fall through
148             }
149 
150             // if not an absolute uri resolve to a file path
151             if (mainUri == null || !mainUri.isAbsolute()) {
152                 File file = new File(mainModuleId);
153                 if (!file.isFile()) {
154                     throw ScriptRuntime.throwError(cx, nativeScope,
155                             "Module \"" + mainModuleId + "\" not found.");
156                 }
157                 mainUri = file.toURI();
158             }
159             mainExports = getExportedModuleInterface(cx, mainUri.toString(),
160                     mainUri, null, true);
161         }
162 
163         this.mainModuleId = mainModuleId;
164         return mainExports;
165     }

176     public Object call(Context cx, Scriptable scope, Scriptable thisObj,
177             Object[] args)
178     {
179         if(args == null || args.length < 1) {
180             throw ScriptRuntime.throwError(cx, scope,
181                     "require() needs one argument");
182         }
183 
184         String id = (String)Context.jsToJava(args[0], String.class);
185         URI uri = null;
186         URI base = null;
187         if (id.startsWith("./") || id.startsWith("../")) {
188             if (!(thisObj instanceof ModuleScope)) {
189                 throw ScriptRuntime.throwError(cx, scope,
190                         "Can't resolve relative module ID \"" + id +
191                                 "\" when require() is used outside of a module");
192             }
193 
194             ModuleScope moduleScope = (ModuleScope) thisObj;
195             base = moduleScope.getBase();
196             URI current = moduleScope.getUri();
197             uri = current.resolve(id);
198 
199             if (base == null) {
200                 // calling module is absolute, resolve to absolute URI
201                 // (but without file extension)
202                 id = uri.toString();
203             } else {
204                 // try to convert to a relative URI rooted on base
205                 id = base.relativize(current).resolve(id).toString();
206                 if (id.charAt(0) == '.') {
207                     // resulting URI is not contained in base,
208                     // throw error or make absolute depending on sandbox flag.
209                     if (sandboxed) {
210                         throw ScriptRuntime.throwError(cx, scope,
211                             "Module \"" + id + "\" is not contained in sandbox.");
212                     } else {
213                         id = uri.toString();
214                     }
215                 }
216             }
217         }
218         return getExportedModuleInterface(cx, id, uri, base, false);
219     }

226     private Scriptable getExportedModuleInterface(Context cx, String id,
227             URI uri, URI base, boolean isMain)
228     {
229         // Check if the requested module is already completely loaded
230         Scriptable exports = exportedModuleInterfaces.get(id);
231         if(exports != null) {
232             if(isMain) {
233                 throw new IllegalStateException(
234                         "Attempt to set main module after it was loaded");
235             }
236             return exports;
237         }
238         // Check if it is currently being loaded on the current thread
239         // (supporting circular dependencies).
240         Map<String, Scriptable> threadLoadingModules =
241             loadingModuleInterfaces.get();
242         if(threadLoadingModules != null) {
243             exports = threadLoadingModules.get(id);
244             if(exports != null) {
245                 return exports;
246             }
247         }
248         // The requested module is neither already loaded, nor is it being
249         // loaded on the current thread. End of fast path. We must synchronize
250         // now, as we have to guarantee that at most one thread can load
251         // modules at any one time. Otherwise, two threads could end up
252         // attempting to load two circularly dependent modules in opposite
253         // order, which would lead to either unacceptable non-determinism or
254         // deadlock, depending on whether we underprotected or overprotected it
255         // with locks.
256         synchronized(loadLock) {
257             // Recheck if it is already loaded - other thread might've
258             // completed loading it just as we entered the synchronized block.
259             exports = exportedModuleInterfaces.get(id);
260             if(exports != null) {
261                 return exports;
262             }
263             // Nope, still not loaded; we're loading it then.
264             final ModuleScript moduleScript = getModule(cx, id, uri, base);
265             if (sandboxed && !moduleScript.isSandboxed()) {
266                 throw ScriptRuntime.throwError(cx, nativeScope, "Module \""
267                         + id + "\" is not contained in sandbox.");
268             }
269             exports = cx.newObject(nativeScope);
270             // Are we the outermost locked invocation on this thread?
271             final boolean outermostLocked = threadLoadingModules == null;
272             if(outermostLocked) {
273                 threadLoadingModules = new HashMap<String, Scriptable>();
274                 loadingModuleInterfaces.set(threadLoadingModules);
275             }
276             // Must make the module exports available immediately on the
277             // current thread, to satisfy the CommonJS Modules/1.1 requirement
278             // that "If there is a dependency cycle, the foreign module may not
279             // have finished executing at the time it is required by one of its
280             // transitive dependencies; in this case, the object returned by
281             // "require" must contain at least the exports that the foreign
282             // module has prepared before the call to require that led to the
283             // current module's execution."
284             threadLoadingModules.put(id, exports);
285             try {
286                 // Support non-standard Node.js feature to allow modules to
287                 // replace the exports object by setting module.exports.
288                 Scriptable newExports = executeModuleScript(cx, id, exports,
289                         moduleScript, isMain);
290                 if (exports != newExports) {
291                     threadLoadingModules.put(id, newExports);
292                     exports = newExports;
293                 }
294             }
295             catch(RuntimeException e) {
296                 // Throw loaded module away if there was an exception
297                 threadLoadingModules.remove(id);
298                 throw e;
299             }
300             finally {
301                 if(outermostLocked) {
302                     // Make loaded modules visible to other threads only after
303                     // the topmost triggering load has completed. This strategy
304                     // (compared to the one where we'd make each module
305                     // globally available as soon as it loads) prevents other
306                     // threads from observing a partially loaded circular
307                     // dependency of a module that completed loading.
308                     exportedModuleInterfaces.putAll(threadLoadingModules);
309                     loadingModuleInterfaces.set(null);
310                 }
311             }
312         }
313         return exports;
314     }

316     private Scriptable executeModuleScript(Context cx, String id,
317             Scriptable exports, ModuleScript moduleScript, boolean isMain)
318     {
319         final ScriptableObject moduleObject = (ScriptableObject)cx.newObject(
320                 nativeScope);
321         URI uri = moduleScript.getUri();
322         URI base = moduleScript.getBase();
323         defineReadOnlyProperty(moduleObject, "id", id);
324         if(!sandboxed) {
325             defineReadOnlyProperty(moduleObject, "uri", uri.toString());
326         }
327         final Scriptable executionScope = new ModuleScope(nativeScope, uri, base);
328         // Set this so it can access the global JS environment objects.
329         // This means we're currently using the "MGN" approach (ModuleScript
330         // with Global Natives) as specified here:
331         // <http://wiki.commonjs.org/wiki/Modules/ProposalForNativeExtension>
332         executionScope.put("exports", executionScope, exports);
333         executionScope.put("module", executionScope, moduleObject);
334         moduleObject.put("exports", moduleObject, exports);
335         install(executionScope);
336         if(isMain) {
337             defineReadOnlyProperty(this, "main", moduleObject);
338         }
339         executeOptionalScript(preExec, cx, executionScope);
340         moduleScript.getScript().exec(cx, executionScope);
341         executeOptionalScript(postExec, cx, executionScope);
342         return ScriptRuntime.toObject(nativeScope,
343                 ScriptableObject.getProperty(moduleObject, "exports"));
344     }

346     private static void executeOptionalScript(Script script, Context cx,
347             Scriptable executionScope)
348     {
349         if(script != null) {

354     private static void defineReadOnlyProperty(ScriptableObject obj,
355             String name, Object value) {
356         ScriptableObject.putProperty(obj, name, value);
357         obj.setAttributes(name, ScriptableObject.READONLY |
358                 ScriptableObject.PERMANENT);
359     }

361     private ModuleScript getModule(Context cx, String id, URI uri, URI base) {
362         try {
363             final ModuleScript moduleScript =
364                     moduleScriptProvider.getModuleScript(cx, id, uri, base, paths);
365             if (moduleScript == null) {
366                 throw ScriptRuntime.throwError(cx, nativeScope, "Module \""
367                         + id + "\" not found.");
368             }
369             return moduleScript;
370         }
371         catch(RuntimeException e) {
372             throw e;
373         }
374         catch(Exception e) {
375             throw Context.throwAsScriptRuntimeEx(e);
376         }
377     }

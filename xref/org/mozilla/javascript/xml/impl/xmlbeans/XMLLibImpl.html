81      XMLName toAttributeName(Context cx, Object nameValue)
82      {
83          String uri;
84          String localName;
85  
86          if (nameValue instanceof String) {
87              uri = "";
88              localName = (String)nameValue;
89          } else if (nameValue instanceof XMLName) {
90              XMLName xmlName = (XMLName)nameValue;
91              if (!xmlName.isAttributeName()) {
92                  xmlName.setAttributeName();
93              }
94              return xmlName;
95          } else if (nameValue instanceof QName) {
96              QName qname = (QName)nameValue;
97              uri = qname.uri();
98              localName = qname.localName();
99          } else if (nameValue instanceof Boolean
100                    || nameValue instanceof Number
101                    || nameValue == Undefined.instance
102                    || nameValue == null)
103         {
104             throw badXMLName(nameValue);
105         } else {
106             uri = "";
107             localName = ScriptRuntime.toString(nameValue);
108         }
109         XMLName xmlName = XMLName.formProperty(uri, localName);
110         xmlName.setAttributeName();
111         return xmlName;
112     }

114     private static RuntimeException badXMLName(Object value)
115     {
116         String msg;
117         if (value instanceof Number) {
118             msg = "Can not construct XML name from number: ";
119         } else if (value instanceof Boolean) {
120             msg = "Can not construct XML name from boolean: ";
121         } else if (value == Undefined.instance || value == null) {
122             msg = "Can not construct XML name from ";
123         } else {
124             throw new IllegalArgumentException(value.toString());
125         }
126         return ScriptRuntime.typeError(msg+ScriptRuntime.toString(value));
127     }

129     XMLName toXMLName(Context cx, Object nameValue)
130     {
131         XMLName result;
132 
133         if (nameValue instanceof XMLName) {
134             result = (XMLName)nameValue;
135         } else if (nameValue instanceof QName) {
136             QName qname = (QName)nameValue;
137             result = XMLName.formProperty(qname.uri(), qname.localName());
138         } else if (nameValue instanceof String) {
139             result = toXMLNameFromString(cx, (String)nameValue);
140         } else if (nameValue instanceof Boolean
141                    || nameValue instanceof Number
142                    || nameValue == Undefined.instance
143                    || nameValue == null)

146         } else {
147             String name = ScriptRuntime.toString(nameValue);
148             result = toXMLNameFromString(cx, name);
149         }

154     /**
155      * If value represents Uint32 index, make it available through
156      * ScriptRuntime.lastUint32Result(cx) and return null.
157      * Otherwise return the same value as toXMLName(cx, value).
158      */
159     XMLName toXMLNameOrIndex(Context cx, Object value)
160     {
161         XMLName result;
162 
163         if (value instanceof XMLName) {
164             result = (XMLName)value;
165         } else if (value instanceof String) {
166             String str = (String)value;
167             long test = ScriptRuntime.testUint32String(str);
168             if (test >= 0) {
169                 ScriptRuntime.storeUint32Result(cx, test);
170                 result = null;
171             } else {
172                 result = toXMLNameFromString(cx, str);
173             }
174         } else if (value instanceof Number) {
175             double d = ((Number)value).doubleValue();
176             long l = (long)d;
177             if (l == d && 0 <= l && l <= 0xFFFFFFFFL) {
178                 ScriptRuntime.storeUint32Result(cx, l);
179                 result = null;
180             } else {
181                 throw badXMLName(value);
182             }
183         } else if (value instanceof QName) {
184             QName qname = (QName)value;
185             String uri = qname.uri();
186             boolean number = false;
187             result = null;
188             if (uri != null && uri.length() == 0) {
189                 // Only in this case qname.toString() can resemble uint32
190                 long test = ScriptRuntime.testUint32String(uri);
191                 if (test >= 0) {
192                     ScriptRuntime.storeUint32Result(cx, test);
193                     number = true;
194                 }
195             }
196             if (!number) {
197                 result = XMLName.formProperty(uri, qname.localName());
198             }
199         } else if (value instanceof Boolean
200                    || value == Undefined.instance
201                    || value == null)
202         {
203             throw badXMLName(value);
204         } else {
205             String str = ScriptRuntime.toString(value);
206             long test = ScriptRuntime.testUint32String(str);
207             if (test >= 0) {
208                 ScriptRuntime.storeUint32Result(cx, test);
209                 result = null;
210             } else {
211                 result = toXMLNameFromString(cx, str);
212             }
213         }
214 
215         return result;
216     }

218     XMLName toXMLNameFromString(Context cx, String name)
219     {
220         if (name == null)
221             throw new IllegalArgumentException();
222 
223         int l = name.length();
224         if (l != 0) {
225             char firstChar = name.charAt(0);
226             if (firstChar == '*') {
227                 if (l == 1) {
228                     return XMLName.formStar();
229                 }
230             } else if (firstChar == '@') {
231                 XMLName xmlName = XMLName.formProperty("", name.substring(1));
232                 xmlName.setAttributeName();
233                 return xmlName;
234             }
235         }
236 
237         String uri = getDefaultNamespaceURI(cx);
238 
239         return XMLName.formProperty(uri, name);
240     }

242     Namespace constructNamespace(Context cx, Object uriValue)
243     {
244         String prefix;
245         String uri;
246 
247         if (uriValue instanceof Namespace) {
248             Namespace ns = (Namespace)uriValue;
249             prefix = ns.prefix();
250             uri = ns.uri();
251         } else if (uriValue instanceof QName) {
252             QName qname = (QName)uriValue;
253             uri = qname.uri();
254             if (uri != null) {
255                 prefix = qname.prefix();
256             } else {
257                 uri = qname.toString();
258                 prefix = null;
259             }
260         } else {
261             uri = ScriptRuntime.toString(uriValue);
262             prefix = (uri.length() == 0) ? "" : null;
263         }
264 
265         return new Namespace(this, prefix, uri);
266     }

281     public Namespace constructNamespace(Context cx, Object prefixValue,
282                                         Object uriValue)
283     {
284         String prefix;
285         String uri;
286 
287         if (uriValue instanceof QName) {
288             QName qname = (QName)uriValue;
289             uri = qname.uri();
290             if (uri == null) {
291                 uri = qname.toString();
292             }
293         } else {
294             uri = ScriptRuntime.toString(uriValue);
295         }
296 
297         if (uri.length() == 0) {
298             if (prefixValue == Undefined.instance) {
299                 prefix = "";
300             } else {
301                 prefix = ScriptRuntime.toString(prefixValue);
302                 if (prefix.length() != 0) {
303                     throw ScriptRuntime.typeError(
304                         "Illegal prefix '"+prefix+"' for 'no namespace'.");
305                 }
306             }
307         } else if (prefixValue == Undefined.instance) {
308             prefix = "";
309         } else if (!isXMLName(cx, prefixValue)) {
310             prefix = "";
311         } else {
312             prefix = ScriptRuntime.toString(prefixValue);
313         }
314 
315         return new Namespace(this, prefix, uri);
316     }

449     Object addXMLObjects(Context cx, XMLObject obj1, XMLObject obj2)
450     {
451         XMLList listToAdd = new XMLList(this);
452 
453         if (obj1 instanceof XMLList) {
454             XMLList list1 = (XMLList)obj1;
455             if (list1.length() == 1) {
456                 listToAdd.addToList(list1.item(0));
457             } else {
458                 // Might be xmlFragment + xmlFragment + xmlFragment + ...;
459                 // then the result will be an XMLList which we want to be an
460                 // rValue and allow it to be assigned to an lvalue.
461                 listToAdd = new XMLList(this, obj1);
462             }
463         } else {
464             listToAdd.addToList(obj1);
465         }
466 
467         if (obj2 instanceof XMLList) {
468             XMLList list2 = (XMLList)obj2;
469             for (int i = 0; i < list2.length(); i++) {
470                 listToAdd.addToList(list2.item(i));
471             }
472         } else if (obj2 instanceof XML) {
473             listToAdd.addToList(obj2);
474         }
475 
476         return listToAdd;
477     }

485     /**
486      * See E4X 13.1.2.1.
487      */
488     public boolean isXMLName(Context cx, Object nameObj)
489     {
490         String name;
491         try {
492             name = ScriptRuntime.toString(nameObj);
493         } catch (EcmaError ee) {
494             if ("TypeError".equals(ee.getName())) {
495                 return false;
496             }
497             throw ee;
498         }
499 
500         // See http://w3.org/TR/xml-names11/#NT-NCName
501         int length = name.length();
502         if (length != 0) {
503             if (isNCNameStartChar(name.charAt(0))) {
504                 for (int i = 1; i != length; ++i) {
505                     if (!isNCNameChar(name.charAt(i))) {
506                         return false;
507                     }
508                 }
509                 return true;
510             }
511         }
512 
513         return false;
514     }

516     private static boolean isNCNameStartChar(int c)
517     {
518         if ((c & ~0x7F) == 0) {
519             // Optimize for ASCII and use A..Z < _ < a..z
520             if (c >= 'a') {
521                 return c <= 'z';
522             } else if (c >= 'A') {
523                 if (c <= 'Z') {
524                     return true;
525                 }
526                 return c == '_';
527             }
528         } else if ((c & ~0x1FFF) == 0) {
529             return (0xC0 <= c && c <= 0xD6)
530                    || (0xD8 <= c && c <= 0xF6)
531                    || (0xF8 <= c && c <= 0x2FF)
532                    || (0x370 <= c && c <= 0x37D)
533                    || 0x37F <= c;
534         }
535         return (0x200C <= c && c <= 0x200D)
536                || (0x2070 <= c && c <= 0x218F)
537                || (0x2C00 <= c && c <= 0x2FEF)
538                || (0x3001 <= c && c <= 0xD7FF)
539                || (0xF900 <= c && c <= 0xFDCF)
540                || (0xFDF0 <= c && c <= 0xFFFD)
541                || (0x10000 <= c && c <= 0xEFFFF);
542     }

544     private static boolean isNCNameChar(int c)
545     {
546         if ((c & ~0x7F) == 0) {
547             // Optimize for ASCII and use - < . < 0..9 < A..Z < _ < a..z
548             if (c >= 'a') {
549                 return c <= 'z';
550             } else if (c >= 'A') {
551                 if (c <= 'Z') {
552                     return true;
553                 }
554                 return c == '_';
555             } else if (c >= '0') {
556                 return c <= '9';
557             } else {
558                 return c == '-' || c == '.';
559             }
560         } else if ((c & ~0x1FFF) == 0) {
561             return isNCNameStartChar(c) || c == 0xB7
562                    || (0x300 <= c && c <= 0x36F);
563         }
564         return isNCNameStartChar(c) || (0x203F <= c && c <= 0x2040);
565     }

630     private Ref xmlPrimaryReference(Context cx, XMLName xmlName,
631                                     Scriptable scope)
632     {
633         XMLObjectImpl xmlObj;
634         XMLObjectImpl firstXmlObject = null;
635         for (;;) {
636             // XML object can only present on scope chain as a wrapper
637             // of XMLWithScope
638             if (scope instanceof XMLWithScope) {
639                 xmlObj = (XMLObjectImpl)scope.getPrototype();
640                 if (xmlObj.hasXMLProperty(xmlName)) {
641                     break;
642                 }
643                 if (firstXmlObject == null) {
644                     firstXmlObject = xmlObj;
645                 }
646             }
647             scope = scope.getParentScope();
648             if (scope == null) {
649                 xmlObj = firstXmlObject;
650                 break;
651             }
652         }
653 
654         // xmlObj == null corresponds to undefined as the target of
655         // the reference
656         if (xmlObj != null) {
657             xmlName.initXMLObject(xmlObj);
658         }
659         return xmlName;
660     }

662     /**
663      * Escapes the reserved characters in a value of an attribute
664      *
665      * @param value Unescaped text
666      * @return The escaped text
667      */
668     public String escapeAttributeValue(Object value)
669     {
670         String text = ScriptRuntime.toString(value);
671 
672         if (text.length() == 0) return "";
673 
674         XmlObject xo = XmlObject.Factory.newInstance();
675 
676         XmlCursor cursor = xo.newCursor();
677         cursor.toNextToken();
678         cursor.beginElement("a");
679         cursor.insertAttributeWithValue("a", text);
680         cursor.dispose();
681 
682         String elementText = xo.toString();
683         int begin = elementText.indexOf('"');
684         int end = elementText.lastIndexOf('"');
685         return elementText.substring(begin + 1, end);
686     }

688     /**
689      * Escapes the reserved characters in a value of a text node
690      *
691      * @param value Unescaped text
692      * @return The escaped text
693      */
694     public String escapeTextValue(Object value)
695     {
696         if (value instanceof XMLObjectImpl) {
697             return ((XMLObjectImpl)value).toXMLString(0);
698         }
699 
700         String text = ScriptRuntime.toString(value);
701 
702         if (text.length() == 0) return text;
703 
704         XmlObject xo = XmlObject.Factory.newInstance();
705 
706         XmlCursor cursor = xo.newCursor();
707         cursor.toNextToken();
708         cursor.beginElement("a");
709         cursor.insertChars(text);
710         cursor.dispose();
711 
712         String elementText = xo.toString();
713         int begin = elementText.indexOf('>') + 1;
714         int end = elementText.lastIndexOf('<');
715         return (begin < end) ? elementText.substring(begin, end) : "";
716     }

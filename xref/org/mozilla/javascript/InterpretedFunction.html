11  final class InterpretedFunction extends NativeFunction implements Script
12  {
13      static final long serialVersionUID = 541475680333911468L;

15      InterpreterData idata;
16      SecurityController securityController;
17      Object securityDomain;
18  
19      private InterpretedFunction(InterpreterData idata,
20                                  Object staticSecurityDomain)
21      {
22          this.idata = idata;
23  
24          // Always get Context from the current thread to
25          // avoid security breaches via passing mangled Context instances
26          // with bogus SecurityController
27          Context cx = Context.getContext();
28          SecurityController sc = cx.getSecurityController();
29          Object dynamicDomain;
30          if (sc != null) {
31              dynamicDomain = sc.getDynamicSecurityDomain(staticSecurityDomain);
32          } else {
33              if (staticSecurityDomain != null) {
34                  throw new IllegalArgumentException();
35              }
36              dynamicDomain = null;
37          }
38  
39          this.securityController = sc;
40          this.securityDomain = dynamicDomain;
41      }
42  
43      private InterpretedFunction(InterpretedFunction parent, int index)
44      {
45          this.idata = parent.idata.itsNestedFunctions[index];
46          this.securityController = parent.securityController;
47          this.securityDomain = parent.securityDomain;
48      }

50      /**
51       * Create script from compiled bytecode.
52       */
53      static InterpretedFunction createScript(InterpreterData idata,
54                                              Object staticSecurityDomain)
55      {
56          InterpretedFunction f;
57          f = new InterpretedFunction(idata, staticSecurityDomain);
58          return f;
59      }

93      /**
94       * Calls the function.
95       * @param cx the current context
96       * @param scope the scope used for the call
97       * @param thisObj the value of "this"
98       * @param args function arguments. Must not be null. You can use
99       * {@link ScriptRuntime#emptyArgs} to pass empty arguments.
100      * @return the result of the function call.
101      */
102     @Override
103     public Object call(Context cx, Scriptable scope, Scriptable thisObj,
104                        Object[] args)
105     {
106         if (!ScriptRuntime.hasTopCall(cx)) {
107             return ScriptRuntime.doTopCall(this, cx, scope, thisObj, args);
108         }
109         return Interpreter.interpret(this, cx, scope, thisObj, args);
110     }
111 
112     public Object exec(Context cx, Scriptable scope)
113     {
114         if (!isScript()) {
115             // Can only be applied to scripts
116             throw new IllegalStateException();
117         }
118         if (!ScriptRuntime.hasTopCall(cx)) {
119             // It will go through "call" path. but they are equivalent
120             return ScriptRuntime.doTopCall(
121                 this, cx, scope, scope, ScriptRuntime.emptyArgs);
122         }
123         return Interpreter.interpret(
124             this, cx, scope, scope, ScriptRuntime.emptyArgs);
125     }

69      public boolean isNumberVar(int varIndex)
70      {
71          varIndex -= fnode.getParamCount();
72          if (varIndex >= 0 && numberVarFlags != null) {
73              return numberVarFlags[varIndex];
74          }
75          return false;

78      void setIsNumberVar(int varIndex)
79      {
80          varIndex -= fnode.getParamCount();
81          // Can only be used with non-parameters
82          if (varIndex < 0) Kit.codeBug();
83          if (numberVarFlags == null) {
84              int size = fnode.getParamAndVarCount() - fnode.getParamCount();
85              numberVarFlags = new boolean[size];
86          }
87          numberVarFlags[varIndex] = true;
88      }

90      public int getVarIndex(Node n)
91      {
92          int index = n.getIntProp(Node.VARIABLE_PROP, -1);
93          if (index == -1) {
94              Node node;
95              int type = n.getType();
96              if (type == Token.GETVAR) {
97                  node = n;
98              } else if (type == Token.SETVAR ||
99                         type == Token.SETCONSTVAR) {
100                 node = n.getFirstChild();
101             } else {
102                 throw Kit.codeBug();
103             }
104             index = fnode.getIndexForNameNode(node);
105             if (index < 0) throw Kit.codeBug();
106             n.putIntProp(Node.VARIABLE_PROP, index);
107         }
108         return index;
109     }

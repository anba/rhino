112     private void exportToScope(boolean sealed) {
113         xmlPrototype = newXML(XmlNode.createText(options, ""));
114         xmlListPrototype = newXMLList();
115         namespacePrototype = Namespace.create(this.globalScope, null,
116                 XmlNode.Namespace.GLOBAL);
117         qnamePrototype = QName.create(this, this.globalScope, null,
118                 XmlNode.QName.create(XmlNode.Namespace.create(""), ""));
119 
120         xmlPrototype.exportAsJSClass(sealed);
121         xmlListPrototype.exportAsJSClass(sealed);
122         namespacePrototype.exportAsJSClass(sealed);
123         qnamePrototype.exportAsJSClass(sealed);
124     }

152     private static RuntimeException badXMLName(Object value)
153     {
154         String msg;
155         if (value instanceof Number) {
156             msg = "Can not construct XML name from number: ";
157         } else if (value instanceof Boolean) {
158             msg = "Can not construct XML name from boolean: ";
159         } else if (value == Undefined.instance || value == null) {
160             msg = "Can not construct XML name from ";
161         } else {
162             throw new IllegalArgumentException(value.toString());
163         }
164         return ScriptRuntime.typeError(msg+ScriptRuntime.toString(value));
165     }

173         XMLName result;

180         } else if (nameValue instanceof String) {

185             || nameValue == null) {

195     /**
196      * If value represents Uint32 index, make it available through
197      * ScriptRuntime.lastUint32Result(cx) and return null.
198      * Otherwise return the same value as toXMLName(cx, value).
199      */
200     XMLName toXMLNameOrIndex(Context cx, Object value)
201     {
202         XMLName result;
203 
204         if (value instanceof XMLName) {
205             result = (XMLName)value;
206         } else if (value instanceof String) {
207             String str = (String)value;
208             long test = ScriptRuntime.testUint32String(str);
209             if (test >= 0) {
210                 ScriptRuntime.storeUint32Result(cx, test);
211                 result = null;
212             } else {
213                 result = toXMLNameFromString(cx, str);
214             }
215         } else if (value instanceof Number) {
216             double d = ((Number)value).doubleValue();
217             long l = (long)d;
218             if (l == d && 0 <= l && l <= 0xFFFFFFFFL) {
219                 ScriptRuntime.storeUint32Result(cx, l);
220                 result = null;
221             } else {
222                 throw badXMLName(value);
223             }
224         } else if (value instanceof QName) {
225             QName qname = (QName)value;
226             String uri = qname.uri();
227             boolean number = false;
228             result = null;
229             if (uri != null && uri.length() == 0) {
230                 // Only in this case qname.toString() can resemble uint32
231                 long test = ScriptRuntime.testUint32String(uri);
232                 if (test >= 0) {
233                     ScriptRuntime.storeUint32Result(cx, test);
234                     number = true;
235                 }
236             }
237             if (!number) {
238                 result = XMLName.formProperty(uri, qname.localName());
239             }
240         } else if (value instanceof Boolean
241                    || value == Undefined.instance
242                    || value == null)
243         {
244             throw badXMLName(value);
245         } else {
246             String str = ScriptRuntime.toString(value);
247             long test = ScriptRuntime.testUint32String(str);
248             if (test >= 0) {
249                 ScriptRuntime.storeUint32Result(cx, test);
250                 result = null;
251             } else {
252                 result = toXMLNameFromString(cx, str);
253             }
254         }
255 
256         return result;
257     }

259     Object addXMLObjects(Context cx, XMLObject obj1, XMLObject obj2)
260     {
261         XMLList listToAdd = newXMLList();
262 
263         if (obj1 instanceof XMLList) {
264             XMLList list1 = (XMLList)obj1;
265             if (list1.length() == 1) {
266                 listToAdd.addToList(list1.item(0));
267             } else {
268                 // Might be xmlFragment + xmlFragment + xmlFragment + ...;
269                 // then the result will be an XMLList which we want to be an
270                 // rValue and allow it to be assigned to an lvalue.
271                 listToAdd = newXMLListFrom(obj1);
272             }
273         } else {
274             listToAdd.addToList(obj1);
275         }
276 
277         if (obj2 instanceof XMLList) {
278             XMLList list2 = (XMLList)obj2;
279             for (int i = 0; i < list2.length(); i++) {
280                 listToAdd.addToList(list2.item(i));
281             }
282         } else if (obj2 instanceof XML) {
283             listToAdd.addToList(obj2);
284         }
285 
286         return listToAdd;
287     }

289     private Ref xmlPrimaryReference(Context cx, XMLName xmlName, Scriptable scope) {
290         XMLObjectImpl xmlObj;
291         XMLObjectImpl firstXml = null;
292         for (;;) {
293             // XML object can only present on scope chain as a wrapper
294             // of XMLWithScope
295             if (scope instanceof XMLWithScope) {
296                 xmlObj = (XMLObjectImpl)scope.getPrototype();
297                 if (xmlObj.hasXMLProperty(xmlName)) {
298                     break;
299                 }
300                 if (firstXml == null) {
301                     firstXml = xmlObj;
302                 }
303             }
304             scope = scope.getParentScope();
305             if (scope == null) {
306                 xmlObj = firstXml;
307                 break;
308             }
309         }
310 
311         // xmlObj == null corresponds to undefined as the target of
312         // the reference
313         if (xmlObj != null) {
314             xmlName.initXMLObject(xmlObj);
315         }
316         return xmlName;
317     }

391     final XML newXMLFromJs(Object inputObject) {
392         String frag;
393 
394         if (inputObject == null || inputObject == Undefined.instance) {
395             frag = "";
396         } else if (inputObject instanceof XMLObjectImpl) {
397             // todo: faster way for XMLObjects?
398             frag = ((XMLObjectImpl) inputObject).toXMLString();
399         } else {
400             frag = ScriptRuntime.toString(inputObject);
401         }
402 
403         if (frag.trim().startsWith("<>")) {
404             throw ScriptRuntime.typeError("Invalid use of XML object anonymous tags <></>.");
405         }
406 
407         if (frag.indexOf("<") == -1) {
408             //    Solo text node
409             return newXML(XmlNode.createText(options, frag));
410         }
411         return parse(frag);
412     }

414     private XML parse(String frag) {
415         try {
416             return newXML(XmlNode.createElement(options,
417                     getDefaultNamespaceURI(Context.getCurrentContext()), frag));
418         } catch (SAXException e) {
419             throw ScriptRuntime.typeError("Cannot parse XML: " + e.getMessage());
420         }

423     final XML ecmaToXml(Object object) {
424         //    See ECMA357 10.3
425         if (object == null || object == Undefined.instance) {
426             throw ScriptRuntime.typeError("Cannot convert " + object + " to XML");
427         }
428         if (object instanceof XML) return (XML)object;
429         if (object instanceof XMLList) {
430             XMLList list = (XMLList)object;
431             if (list.getXML() != null) {
432                 return list.getXML();
433             } else {
434                 throw ScriptRuntime.typeError("Cannot convert list of >1 element to XML");
435             }
436         }
437         //    TODO    Technically we should fail on anything except a String, Number or Boolean
438         //            See ECMA357 10.3
439         // Extension: if object is a DOM node, use that to construct the XML
440         // object.
441         if (object instanceof Wrapper) {
442             object = ((Wrapper) object).unwrap();
443         }
444         if (object instanceof org.w3c.dom.Node) {
445             org.w3c.dom.Node node = (org.w3c.dom.Node) object;
446             return newXML(XmlNode.createElementFromNode(node));
447         }
448         //    Instead we just blindly cast to a String and let them convert anything.
449         String s = ScriptRuntime.toString(object);
450         //    TODO    Could this get any uglier?
451         if (s.length() > 0 && s.charAt(0) == '<') {
452             return parse(s);
453         } else {
454             return newXML(XmlNode.createText(options, s));
455         }
456     }

466     final XMLList newXMLListFrom(Object inputObject) {
467         XMLList rv = newXMLList();
468 
469         if (inputObject == null || inputObject instanceof Undefined) {
470             return rv;
471         } else if (inputObject instanceof XML) {
472             XML xml = (XML) inputObject;
473             rv.getNodeList().add(xml);
474             return rv;
475         } else if (inputObject instanceof XMLList) {
476             XMLList xmll = (XMLList) inputObject;
477             rv.getNodeList().add(xmll.getNodeList());
478             return rv;
479         } else {
480             String frag = ScriptRuntime.toString(inputObject).trim();
481 
482             if (!frag.startsWith("<>")) {
483                 frag = "<>" + frag + "</>";
484             }
485 
486             frag = "<fragment>" + frag.substring(2);
487             if (!frag.endsWith("</>")) {
488                 throw ScriptRuntime.typeError("XML with anonymous tag missing end anonymous tag");
489             }
490 
491             frag = frag.substring(0, frag.length() - 3) + "</fragment>";
492 
493             XML orgXML = newXMLFromJs(frag);
494 
495             // Now orphan the children and add them to our XMLList.
496             XMLList children = orgXML.children();
497 
498             for (int i = 0; i < children.getNodeList().length(); i++) {
499                 // Copy here is so that they'll be orphaned (parent() will be undefined)
500                 rv.getNodeList().add(((XML) children.item(i).copy()));
501             }
502             return rv;
503         }
504     }

506     XmlNode.QName toNodeQName(Context cx, Object namespaceValue, Object nameValue) {
507         // This is duplication of constructQName(cx, namespaceValue, nameValue)
508         // but for XMLName
509 
510         String localName;
511 
512         if (nameValue instanceof QName) {
513             QName qname = (QName)nameValue;
514             localName = qname.localName();
515         } else {
516             localName = ScriptRuntime.toString(nameValue);
517         }
518 
519         XmlNode.Namespace ns;
520         if (namespaceValue == Undefined.instance) {
521             if ("*".equals(localName)) {
522                 ns = null;
523             } else {
524                 ns = getDefaultNamespace(cx).getDelegate();
525             }
526         } else if (namespaceValue == null) {
527             ns = null;
528         } else if (namespaceValue instanceof Namespace) {
529             ns = ((Namespace)namespaceValue).getDelegate();
530         } else {
531             ns = this.namespacePrototype.constructNamespace(namespaceValue).getDelegate();
532         }
533 
534         if (localName != null && localName.equals("*")) localName = null;
535         return XmlNode.QName.create(ns, localName);
536     }

538     XmlNode.QName toNodeQName(Context cx, String name, boolean attribute) {
539         XmlNode.Namespace defaultNamespace = getDefaultNamespace(cx).getDelegate();
540         if (name != null && name.equals("*")) {
541             return XmlNode.QName.create(null, null);
542         } else {
543             if (attribute) {
544                 return XmlNode.QName.create(XmlNode.Namespace.GLOBAL, name);
545             } else {
546                 return XmlNode.QName.create(defaultNamespace, name);
547             }
548         }
549     }

555     XmlNode.QName toNodeQName(Context cx, Object nameValue, boolean attribute) {
556         if (nameValue instanceof XMLName) {
557             return ((XMLName)nameValue).toQname();
558         } else if (nameValue instanceof QName) {
559             QName qname = (QName)nameValue;
560             return qname.getDelegate();
561         } else if (
562             nameValue instanceof Boolean
563             || nameValue instanceof Number
564             || nameValue == Undefined.instance
565             || nameValue == null
566         ) {
567             throw badXMLName(nameValue);
568         } else {
569             String local = null;
570             if (nameValue instanceof String) {
571                 local = (String)nameValue;
572             } else {
573                 local = ScriptRuntime.toString(nameValue);
574             }
575             return toNodeQName(cx, local, attribute);

131     private String toXmlNewlines(String rv) {
132         StringBuffer nl = new StringBuffer();
133         for (int i=0; i<rv.length(); i++) {
134             if (rv.charAt(i) == '\r') {
135                 if (rv.charAt(i+1) == '\n') {
136                     //    DOS, do nothing and skip the \r
137                 } else {
138                     //    Macintosh, substitute \n
139                     nl.append('\n');
140                 }
141             } else {
142                 nl.append(rv.charAt(i));
143             }
144         }
145         return nl.toString();
146     }

218     final Node toXml(String defaultNamespaceUri, String xml) throws org.xml.sax.SAXException {
219         //    See ECMA357 10.3.1
220         DocumentBuilder builder = null;
221         try {
222             String syntheticXml = "<parent xmlns=\"" + defaultNamespaceUri +
223                 "\">" + xml + "</parent>";
224             builder = getDocumentBuilderFromPool();
225             Document document = builder.parse( new org.xml.sax.InputSource(new java.io.StringReader(syntheticXml)) );
226             if (ignoreProcessingInstructions) {
227                 List<Node> list = new java.util.ArrayList<Node>();
228                 addProcessingInstructionsTo(list, document);
229                 for (Node node: list) {
230                     node.getParentNode().removeChild(node);
231                 }
232             }
233             if (ignoreComments) {
234                 List<Node> list = new java.util.ArrayList<Node>();
235                 addCommentsTo(list, document);
236                 for (Node node: list) {
237                     node.getParentNode().removeChild(node);
238                 }
239             }
240             if (ignoreWhitespace) {
241                 //    Apparently JAXP setIgnoringElementContentWhitespace() has a different meaning, it appears from the Javadoc
242                 //    Refers to element-only content models, which means we would need to have a validating parser and DTD or schema
243                 //    so that it would know which whitespace to ignore.
244 
245                 //    Instead we will try to delete it ourselves.
246                 List<Node> list = new java.util.ArrayList<Node>();
247                 addTextNodesToRemoveAndTrim(list, document);
248                 for (Node node: list) {
249                     node.getParentNode().removeChild(node);
250                 }
251             }
252             NodeList rv = document.getDocumentElement().getChildNodes();
253             if (rv.getLength() > 1) {
254                 throw ScriptRuntime.constructError("SyntaxError", "XML objects may contain at most one node.");
255             } else if (rv.getLength() == 0) {
256                 Node node = document.createTextNode("");
257                 return node;
258             } else {
259                 Node node = rv.item(0);
260                 document.getDocumentElement().removeChild(node);
261                 return node;
262             }
263         } catch (java.io.IOException e) {
264             throw new RuntimeException("Unreachable.");
265         } catch (javax.xml.parsers.ParserConfigurationException e) {
266             throw new RuntimeException(e);
267         } finally {
268             if (builder != null)
269                 returnDocumentBuilderToPool(builder);
270         }
271     }

309     String escapeAttributeValue(Object value) {
310         String text = ScriptRuntime.toString(value);
311 
312         if (text.length() == 0) return "";
313 
314         Document dom = newDocument();
315         Element e = dom.createElement("a");
316         e.setAttribute("b", text);
317         String elementText = toString(e);
318         int begin = elementText.indexOf('"');
319         int end = elementText.lastIndexOf('"');
320         return elementText.substring(begin+1,end);
321     }

324         if (value instanceof XMLObjectImpl) {

334         e.setTextContent(text);
335         String elementText = toString(e);

356     final String ecmaToXmlString(Node node) {
357         //    See ECMA 357 Section 10.2.1
358         StringBuffer s = new StringBuffer();
359         int indentLevel = 0;
360         if (prettyPrint) {
361             for (int i=0; i<indentLevel; i++) {
362                 s.append(' ');
363             }
364         }
365         if (node instanceof Text) {
366             String data = ((Text)node).getData();
367             //    TODO Does Java trim() work same as XMLWhitespace?
368             String v = (prettyPrint) ? data.trim() : data;
369             s.append(escapeElementValue(v));
370             return s.toString();
371         }
372         if (node instanceof Attr) {
373             String value = ((Attr)node).getValue();
374             s.append(escapeAttributeValue(value));
375             return s.toString();
376         }
377         if (node instanceof Comment) {
378             s.append("<!--" + ((Comment)node).getNodeValue() + "-->");
379             return s.toString();
380         }
381         if (node instanceof ProcessingInstruction) {
382             ProcessingInstruction pi = (ProcessingInstruction)node;
383             s.append("<?" + pi.getTarget() + " " + pi.getData() + "?>");
384             return s.toString();
385         }
386         s.append(elementToXmlString((Element)node));
387         return s.toString();
388     }

390     private void beautifyElement(Element e, int indent) {
391         StringBuffer s = new StringBuffer();
392         s.append('\n');
393         for (int i=0; i<indent; i++) {
394             s.append(' ');
395         }
396         String afterContent = s.toString();
397         for (int i=0; i<prettyIndent; i++) {
398             s.append(' ');
399         }
400         String beforeContent = s.toString();
401 
402         //    We "mark" all the nodes first; if we tried to do this loop otherwise, it would behave unexpectedly (the inserted nodes
403         //    would contribute to the length and it might never terminate).
404         ArrayList<Node> toIndent = new ArrayList<Node>();
405         boolean indentChildren = false;
406         for (int i=0; i<e.getChildNodes().getLength(); i++) {
407             if (i == 1) indentChildren = true;
408             if (e.getChildNodes().item(i) instanceof Text) {
409                 toIndent.add(e.getChildNodes().item(i));
410             } else {
411                 indentChildren = true;
412                 toIndent.add(e.getChildNodes().item(i));
413             }
414         }
415         if (indentChildren) {
416             for (int i=0; i<toIndent.size(); i++) {
417                 e.insertBefore(e.getOwnerDocument().createTextNode(beforeContent),
418                         toIndent.get(i));
419             }
420         }
421         NodeList nodes = e.getChildNodes();
422         ArrayList<Element> list = new ArrayList<Element>();
423         for (int i=0; i < nodes.getLength(); i++) {
424             if (nodes.item(i) instanceof Element) {
425                 list.add((Element)nodes.item(i));
426             }
427         }
428         for (Element elem: list) {
429             beautifyElement(elem, indent + prettyIndent);
430         }
431         if (indentChildren) {
432             e.appendChild(e.getOwnerDocument().createTextNode(afterContent));
433         }

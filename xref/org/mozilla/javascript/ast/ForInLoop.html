121     @Override
122     public String toSource(int depth) {
123         StringBuilder sb = new StringBuilder();
124         sb.append(makeIndent(depth));
125         sb.append("for ");
126         if (isForEach()) {
127             sb.append("each ");
128         }
129         sb.append("(");
130         sb.append(iterator.toSource(0));
131         sb.append(" in ");
132         sb.append(iteratedObject.toSource(0));
133         sb.append(") ");
134         if (body instanceof Block) {
135             sb.append(body.toSource(depth).trim()).append("\n");
136         } else {
137             sb.append("\n").append(body.toSource(depth+1));
138         }
139         return sb.toString();
140     }

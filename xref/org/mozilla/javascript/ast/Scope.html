76       * Sets the child's parent scope to this scope.
77       * @throws IllegalStateException if the child's parent scope is
78       * non-{@code null}
79       */
80      public void addChildScope(Scope child) {
81          if (childScopes == null) {
82              childScopes = new ArrayList<Scope>();
83          }
84          childScopes.add(child);

86      }

88      /**
89       * Used by the parser; not intended for typical use.
90       * Changes the parent-scope links for this scope's child scopes
91       * to the specified new scope.  Copies symbols from this scope
92       * into new scope.
93       *
94       * @param newScope the scope that will replace this one on the
95       *        scope stack.
96       */
97      public void replaceWith(Scope newScope) {
98          if (childScopes != null) {
99              for (Scope kid : childScopes) {
100                 newScope.addChildScope(kid);  // sets kid's parent
101             }
102             childScopes.clear();
103             childScopes = null;
104         }
105         if (symbolTable != null && !symbolTable.isEmpty()) {
106             joinScopes(this, newScope);
107         }
108     }

124     /**
125      * Creates a new scope node, moving symbol table information
126      * from "scope" to the new node, and making "scope" a nested
127      * scope contained by the new node.
128      * Useful for injecting a new scope in a scope chain.
129      */
130     public static Scope splitScope(Scope scope) {
131         Scope result = new Scope(scope.getType());
132         result.symbolTable = scope.symbolTable;
133         scope.symbolTable = null;
134         result.parent = scope.parent;
135         result.setParentScope(scope.getParentScope());
136         result.setParentScope(result);
137         scope.parent = result;
138         result.top = scope.top;
139         return result;
140     }

142     /**
143      * Copies all symbols from source scope to dest scope.
144      */
145     public static void joinScopes(Scope source, Scope dest) {
146         Map<String,Symbol> src = source.ensureSymbolTable();
147         Map<String,Symbol> dst = dest.ensureSymbolTable();
148         if (!Collections.disjoint(src.keySet(), dst.keySet())) {
149             codeBug();
150         }
151         for (Map.Entry<String, Symbol> entry: src.entrySet()) {
152             Symbol sym = entry.getValue();
153             sym.setContainingTable(dest);
154             dst.put(entry.getKey(), sym);
155         }
156     }

158     /**
159      * Returns the scope in which this name is defined
160      * @param name the symbol to look up
161      * @return this {@link Scope}, one of its parent scopes, or {@code null} if
162      * the name is not defined any this scope chain
163      */
164     public Scope getDefiningScope(String name) {
165         for (Scope s = this; s != null; s = s.parentScope) {
166             Map<String,Symbol> symbolTable = s.getSymbolTable();
167             if (symbolTable != null && symbolTable.containsKey(name)) {
168                 return s;
169             }
170         }
171         return null;
172     }

186     public void putSymbol(Symbol symbol) {
187         if (symbol.getName() == null)
188             throw new IllegalArgumentException("null symbol name");
189         ensureSymbolTable();
190         symbolTable.put(symbol.getName(), symbol);
191         symbol.setContainingTable(this);
192         top.addSymbol(symbol);

210     private Map<String,Symbol> ensureSymbolTable() {
211         if (symbolTable == null) {
212             symbolTable = new LinkedHashMap<String,Symbol>(5);
213         }

217     /**
218      * Returns a copy of the child list, with each child cast to an
219      * {@link AstNode}.
220      * @throws ClassCastException if any non-{@code AstNode} objects are
221      * in the child list, e.g. if this method is called after the code
222      * generator begins the tree transformation.
223      */
224     public List<AstNode> getStatements() {
225         List<AstNode> stmts = new ArrayList<AstNode>();
226         Node n = getFirstChild();
227         while (n != null) {
228             stmts.add((AstNode)n);
229             n = n.getNext();
230         }
231         return stmts;
232     }

234     @Override
235     public String toSource(int depth) {
236         StringBuilder sb = new StringBuilder();
237         sb.append(makeIndent(depth));
238         sb.append("{\n");
239         for (Node kid : this) {
240             sb.append(((AstNode)kid).toSource(depth+1));
241         }
242         sb.append(makeIndent(depth));
243         sb.append("}\n");
244         return sb.toString();

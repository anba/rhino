15  /**
16   * Try/catch/finally statement.  Node type is {@link Token#TRY}.<p>
17   *
18   * <pre><i>TryStatement</i> :
19   *        <b>try</b> Block Catch
20   *        <b>try</b> Block Finally
21   *        <b>try</b> Block Catch Finally
22   * <i>Catch</i> :
23   *        <b>catch</b> ( <i><b>Identifier</b></i> ) Block
24   * <i>Finally</i> :
25   *        <b>finally</b> Block</pre>

34      private AstNode finallyBlock;
35      private int finallyPosition = -1;

66      /**
67       * Returns list of {@link CatchClause} nodes.  If there are no catch
68       * clauses, returns an immutable empty list.
69       */
70      public List<CatchClause> getCatchClauses() {
71          return catchClauses != null ? catchClauses : NO_CATCHES;
72      }
73  
74      /**
75       * Sets list of {@link CatchClause} nodes.  Also sets their parents
76       * to this node.  May be {@code null}.  Replaces any existing catch
77       * clauses for this node.
78       */
79      public void setCatchClauses(List<CatchClause> catchClauses) {
80          if (catchClauses == null) {
81              this.catchClauses = null;
82          } else {
83              if (this.catchClauses != null)
84                  this.catchClauses.clear();
85              for (CatchClause cc : catchClauses) {
86                  addCatchClause(cc);
87              }
88          }
89      }
90  
91      /**
92       * Add a catch-clause to the end of the list, and sets its parent to
93       * this node.
94       * @throws IllegalArgumentException} if {@code clause} is {@code null}
95       */
96      public void addCatchClause(CatchClause clause) {
97          assertNotNull(clause);
98          if (catchClauses == null) {
99              catchClauses = new ArrayList<CatchClause>();
100         }
101         catchClauses.add(clause);
102         clause.setParent(this);
103     }

136     @Override
137     public String toSource(int depth) {
138         StringBuilder sb = new StringBuilder(250);
139         sb.append(makeIndent(depth));
140         sb.append("try ");
141         sb.append(tryBlock.toSource(depth).trim());
142         for (CatchClause cc : getCatchClauses()) {
143             sb.append(cc.toSource(depth));
144         }
145         if (finallyBlock != null) {
146             sb.append(" finally ");
147             sb.append(finallyBlock.toSource(depth));
148         }
149         return sb.toString();
150     }

152     /**
153      * Visits this node, then the try-block, then any catch clauses,
154      * and then any finally block.
155      */
156     @Override
157     public void visit(NodeVisitor v) {
158         if (v.visit(this)) {
159             tryBlock.visit(v);
160             for (CatchClause cc : getCatchClauses()) {
161                 cc.visit(v);
162             }
163             if (finallyBlock != null) {
164                 finallyBlock.visit(v);
165             }
166         }

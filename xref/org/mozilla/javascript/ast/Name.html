7   package org.mozilla.javascript.ast;

9   import org.mozilla.javascript.Token;

11  /**
12   * AST node for a simple name.  A simple name is an identifier that is
13   * not a keyword. Node type is {@link Token#NAME}.<p>
14   *
15   * This node type is also used to represent certain non-identifier names that
16   * are part of the language syntax.  It's used for the "get" and "set"
17   * pseudo-keywords for object-initializer getter/setter properties, and it's
18   * also used for the "*" wildcard in E4X XML namespace and name expressions.
19   */
20  public class Name extends AstNode {
21  
22      private String identifier;
23      private Scope scope;
24  
25      {
26          type = Token.NAME;
27      }
28  
29      public Name() {
30      }
31  
32      public Name(int pos) {
33          super(pos);
34      }
35  
36      public Name(int pos, int len) {
37          super(pos, len);
38      }
39  
40      /**
41       * Constructs a new {@link Name}
42       * @param pos node start position
43       * @param len node length
44       * @param name the identifier associated with this {@code Name} node
45       */
46      public Name(int pos, int len, String name) {
47          super(pos, len);
48          setIdentifier(name);
49      }
50  
51      public Name(int pos, String name) {
52          super(pos);
53          setIdentifier(name);
54          setLength(name.length());
55      }
56  
57      /**
58       * Returns the node's identifier
59       */
60      public String getIdentifier() {
61          return identifier;
62      }
63  
64      /**
65       * Sets the node's identifier
66       * @throws IllegalArgumentException if identifier is null
67       */
68      public void setIdentifier(String identifier) {
69          assertNotNull(identifier);
70          this.identifier = identifier;
71          setLength(identifier.length());
72      }
73  
74      /**
75       * Set the {@link Scope} associated with this node.  This method does not
76       * set the scope's ast-node field to this node.  The field exists only
77       * for temporary storage by the code generator.  Not every name has an
78       * associated scope - typically only function and variable names (but not
79       * property names) are registered in a scope.
80       *
81       * @param s the scope.  Can be null.  Doesn't set any fields in the
82       * scope.
83       */
84      public void setScope(Scope s) {
85          scope = s;
86      }
87  
88      /**
89       * Return the {@link Scope} associated with this node.  This is
90       * <em>only</em> used for (and set by) the code generator, so it will always
91       * be null in frontend AST-processing code.  Use {@link #getDefiningScope}
92       * to find the lexical {@code Scope} in which this {@code Name} is defined,
93       * if any.
94       */
95      public Scope getScope() {
96          return scope;
97      }
98  
99      /**
100      * Returns the {@link Scope} in which this {@code Name} is defined.
101      * @return the scope in which this name is defined, or {@code null}
102      * if it's not defined in the current lexical scope chain
103      */
104     public Scope getDefiningScope() {
105         Scope enclosing = getEnclosingScope();
106         String name = getIdentifier();
107         return enclosing == null ? null : enclosing.getDefiningScope(name);
108     }
109 
110     /**
111      * Return true if this node is known to be defined as a symbol in a
112      * lexical scope other than the top-level (global) scope.
113      *
114      * @return {@code true} if this name appears as local variable, a let-bound
115      * variable not in the global scope, a function parameter, a loop
116      * variable, the property named in a {@link PropertyGet}, or in any other
117      * context where the node is known not to resolve to the global scope.
118      * Returns {@code false} if the node is defined in the top-level scope
119      * (i.e., its defining scope is an {@link AstRoot} object), or if its
120      * name is not defined as a symbol in the symbol table, in which case it
121      * may be an external or built-in name (or just an error of some sort.)
122      */
123     public boolean isLocalName() {
124         Scope scope = getDefiningScope();
125         return scope != null && scope.getParentScope() != null;
126     }
127 
128     /**
129      * Return the length of this node's identifier, to let you pretend
130      * it's a {@link String}.  Don't confuse this method with the
131      * {@link AstNode#getLength} method, which returns the range of
132      * characters that this node overlaps in the source input.
133      */
134     public int length() {
135         return identifier == null ? 0 : identifier.length();
136     }
137 
138     @Override
139     public String toSource(int depth) {
140         return makeIndent(depth) + (identifier == null ? "<null>" : identifier);
141     }

143     /**
144      * Visits this node.  There are no children to visit.
145      */
146     @Override
147     public void visit(NodeVisitor v) {
148         v.visit(this);
149     }

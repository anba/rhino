19  /**
20   * Base class for AST node types.  The goal of the AST is to represent the
21   * physical source code, to make it useful for code-processing tools such
22   * as IDEs or pretty-printers.  The parser must not rewrite the parse tree
23   * when producing this representation. <p>
24   *
25   * The {@code AstNode} hierarchy sits atop the older {@link Node} class,
26   * which was designed for code generation.  The {@code Node} class is a
27   * flexible, weakly-typed class suitable for creating and rewriting code
28   * trees, but using it requires you to remember the exact ordering of the
29   * child nodes, which are kept in a linked list.  The {@code AstNode}
30   * hierarchy is a strongly-typed facade with named accessors for children
31   * and common properties, but under the hood it's still using a linked list
32   * of child nodes.  It isn't a very good idea to use the child list directly
33   * unless you know exactly what you're doing.</p>
34   *
35   * Note that {@code AstNode} records additional information, including
36   * the node's position, length, and parent node.  Also, some {@code AstNode}
37   * subclasses record some of their child nodes in instance members, since
38   * they are not needed for code generation.  In a nutshell, only the code
39   * generator should be mixing and matching {@code AstNode} and {@code Node}
40   * objects.<p>
41   *
42   * All offset fields in all subclasses of AstNode are relative to their
43   * parent.  For things like paren, bracket and keyword positions, the
44   * position is relative to the current node.  The node start position is
45   * relative to the parent node. <p>
46   *
47   * During the actual parsing, node positions are absolute; adding the node to
48   * its parent fixes up the offsets to be relative.  By the time you see the AST
49   * (e.g. using the {@code Visitor} interface), the offsets are relative. <p>
50   *
51   * {@code AstNode} objects have property lists accessible via the
52   * {@link #getProp} and {@link #putProp} methods.  The property lists are
53   * integer-keyed with arbitrary {@code Object} values.  For the most part the
54   * parser generating the AST avoids using properties, preferring fields for
55   * elements that are always set.  Property lists are intended for user-defined
56   * annotations to the tree.  The Rhino code generator acts as a client and
57   * uses node properties extensively.  You are welcome to use the property-list
58   * API for anything your client needs.<p>
59   *
60   * This hierarchy does not have separate branches for expressions and
61   * statements, as the distinction in JavaScript is not as clear-cut as in
62   * Java or C++. <p>
63   */

232      * current node's start position to be relative to the new parent.

235     public void setParent(AstNode parent) {
236         if (parent == this.parent) {

238         }
239 
240         // Convert position back to absolute.
241         if (this.parent != null) {
242             setRelative(-this.parent.getPosition());
243         }

246         if (parent != null) {
247             setRelative(parent.getPosition());
248         }

263         addChildToBack(kid);

444     /**
445      * Prints a comma-separated item list into a {@link StringBuilder}.
446      * @param items a list to print
447      * @param sb a {@link StringBuilder} into which to print
448      */
449     protected <T extends AstNode> void printList(List<T> items,
450                                                  StringBuilder sb) {
451         int max = items.size();
452         int count = 0;
453         for (AstNode item : items) {
454             sb.append(item.toSource(0));
455             if (count++ < max-1) {
456                 sb.append(", ");
457             } else if (item instanceof EmptyExpression) {
458                 sb.append(",");
459             }
460         }
461     }

512     /**
513      * Permits AST nodes to be sorted based on start position and length.
514      * This makes it easy to sort Comment and Error nodes into a set of
515      * other AST nodes:  just put them all into a {@link java.util.SortedSet},
516      * for instance.
517      * @param other another node
518      * @return -1 if this node's start position is less than {@code other}'s
519      * start position.  If tied, -1 if this node's length is less than
520      * {@code other}'s length.  If the lengths are equal, sorts abitrarily
521      * on hashcode unless the nodes are the same per {@link #equals}.
522      */
523     public int compareTo(AstNode other) {
524         if (this.equals(other)) return 0;
525         int abs1 = this.getAbsolutePosition();
526         int abs2 = other.getAbsolutePosition();
527         if (abs1 < abs2) return -1;
528         if (abs2 < abs1) return 1;
529         int len1 = this.getLength();
530         int len2 = other.getLength();
531         if (len1 < len2) return -1;
532         if (len2 < len1) return 1;
533         return this.hashCode() - other.hashCode();
534     }

545     protected static class DebugPrintVisitor implements NodeVisitor {
546         private StringBuilder buffer;
547         private static final int DEBUG_INDENT = 2;
548         public DebugPrintVisitor(StringBuilder buf) {
549             buffer = buf;
550         }
551         public String toString() {
552             return buffer.toString();
553         }
554         private String makeIndent(int depth) {
555             StringBuilder sb = new StringBuilder(DEBUG_INDENT * depth);
556             for (int i = 0; i < (DEBUG_INDENT * depth); i++) {
557                 sb.append(" ");
558             }
559             return sb.toString();
560         }
561         public boolean visit(AstNode node) {
562             int tt = node.getType();
563             String name = Token.typeToName(tt);
564             buffer.append(node.getAbsolutePosition()).append("\t");
565             buffer.append(makeIndent(node.depth()));
566             buffer.append(name).append(" ");
567             buffer.append(node.getPosition()).append(" ");
568             buffer.append(node.getLength());
569             if (tt == Token.NAME) {
570                 buffer.append(" ").append(((Name)node).getIdentifier());
571             }
572             buffer.append("\n");
573             return true;  // process kids
574         }
575     }

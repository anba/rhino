11  /**
12   * AST node for a single 'for (foo in bar)' loop construct in a JavaScript 1.7
13   * Array comprehension.  This node type is almost equivalent to a
14   * {@link ForInLoop}, except that it has no body statement.
15   * Node type is {@link Token#FOR}.<p>
16   */
17  public class ArrayComprehensionLoop extends ForInLoop {
18  
19      public ArrayComprehensionLoop() {
20      }
21  
22      public ArrayComprehensionLoop(int pos) {
23          super(pos);
24      }
25  
26      public ArrayComprehensionLoop(int pos, int len) {
27          super(pos, len);
28      }
29      
30      /**
31       * Returns {@code null} for loop body
32       * @return loop body (always {@code null} for this node type)
33       */
34      public AstNode getBody() {
35          return null;
36      }
37  
38      /**
39       * Throws an exception on attempts to set the loop body.
40       * @param body loop body
41       * @throws UnsupportedOperationException
42       */
43      public void setBody(AstNode body) {
44          throw new UnsupportedOperationException("this node type has no body");
45      }
46  
47      @Override
48      public String toSource(int depth) {
49          return makeIndent(depth)
50                  + " for "
51                  + (isForEach()?"each ":"")
52                  + "("
53                  + iterator.toSource(0)
54                  + " in "
55                  + iteratedObject.toSource(0)
56                  + ")";
57      }
58  
59      /**
60       * Visits the iterator expression and the iterated object expression.
61       * There is no body-expression for this loop type.
62       */
63      @Override
64      public void visit(NodeVisitor v) {
65          if (v.visit(this)) {
66              iterator.visit(v);
67              iteratedObject.visit(v);
68          }
69      }

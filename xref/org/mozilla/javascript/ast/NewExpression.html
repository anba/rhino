42       * Returns initializer object, if any.
43       * @return extra initializer object-literal expression, or {@code null} if
44       * not specified.

50      /**
51       * Sets initializer object.  Rhino supports an experimental syntax
52       * of the form {@code new expr [ ( arglist ) ] [initializer]},
53       * in which initializer is an object literal that is used to set
54       * additional properties on the newly-created {@code expr} object.
55       *
56       * @param initializer extra initializer object.
57       * Can be {@code null}.
58       */
59      public void setInitializer(ObjectLiteral initializer) {
60        this.initializer = initializer;
61        if (initializer != null)
62            initializer.setParent(this);
63      }

65      @Override
66      public String toSource(int depth) {
67          StringBuilder sb = new StringBuilder();
68          sb.append(makeIndent(depth));
69          sb.append("new ");
70          sb.append(target.toSource(0));
71          sb.append("(");
72          if (arguments != null) {
73              printList(arguments, sb);
74          }
75          sb.append(")");
76          if (initializer != null) {
77              sb.append(" ");
78              sb.append(initializer.toSource(0));
79          }
80          return sb.toString();
81      }

83      /**
84       * Visits this node, the target, and each argument.  If there is
85       * a trailing initializer node, visits that last.
86       */
87      @Override
88      public void visit(NodeVisitor v) {
89          if (v.visit(this)) {
90              target.visit(v);
91              for (AstNode arg : getArguments()) {
92                  arg.visit(v);
93              }
94              if (initializer != null) {

96              }
97          }

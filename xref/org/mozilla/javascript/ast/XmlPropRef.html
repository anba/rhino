11  /**
12   * AST node for an E4X XML {@code [expr]} property-ref expression.
13   * The node type is {@link Token#REF_NAME}.<p>
14   *
15   * Syntax:<p>
16   *
17   * <pre> @<i><sub>opt</sub></i> ns:: <i><sub>opt</sub></i> name</pre>
18   *
19   * Examples include {@code name}, {@code ns::name}, {@code ns::*},
20   * {@code *::name}, {@code *::*}, {@code @attr}, {@code @ns::attr},
21   * {@code @ns::*}, {@code @*::attr}, {@code @*::*} and {@code @*}.<p>
22   *
23   * The node starts at the {@code @} token, if present.  Otherwise it starts
24   * at the namespace name.  The node bounds extend through the closing
25   * right-bracket, or if it is missing due to a syntax error, through the
26   * end of the index expression.<p>
27   */
28  public class XmlPropRef extends XmlRef {

64      @Override
65      public String toSource(int depth) {
66          StringBuilder sb = new StringBuilder();
67          sb.append(makeIndent(depth));
68          if (isAttributeAccess()) {
69              sb.append("@");

71          if (namespace != null) {
72              sb.append(namespace.toSource(0));
73              sb.append("::");
74          }
75          sb.append(propName.toSource(0));

77      }

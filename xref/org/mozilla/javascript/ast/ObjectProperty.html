7   package org.mozilla.javascript.ast;

9   import org.mozilla.javascript.Token;
10  
11  /**
12   * AST node for a single name:value entry in an Object literal.
13   * For simple entries, the node type is {@link Token#COLON}, and
14   * the name (left side expression) is either a {@link Name}, a
15   * {@link StringLiteral} or a {@link NumberLiteral}.<p>
16   *
17   * This node type is also used for getter/setter properties in object
18   * literals.  In this case the node bounds include the "get" or "set"
19   * keyword.  The left-hand expression in this case is always a
20   * {@link Name}, and the overall node type is {@link Token#GET} or
21   * {@link Token#SET}, as appropriate.<p>
22   *
23   * The {@code operatorPosition} field is meaningless if the node is
24   * a getter or setter.<p>
25   *
26   * <pre><i>ObjectProperty</i> :
27   *       PropertyName <b>:</b> AssignmentExpression
28   * <i>PropertyName</i> :
29   *       Identifier
30   *       StringLiteral
31   *       NumberLiteral</pre>
32   */
33  public class ObjectProperty extends InfixExpression {
34  
35      {
36          type = Token.COLON;
37      }
38  
39      /**
40       * Sets the node type.  Must be one of
41       * {@link Token#COLON}, {@link Token#GET}, or {@link Token#SET}.
42       * @throws IllegalArgumentException if {@code nodeType} is invalid
43       */
44      public void setNodeType(int nodeType) {
45          if (nodeType != Token.COLON
46              && nodeType != Token.GET
47              && nodeType != Token.SET)
48              throw new IllegalArgumentException("invalid node type: "
49                                                 + nodeType);
50          setType(nodeType);
51      }
52  
53      public ObjectProperty() {
54      }
55  
56      public ObjectProperty(int pos) {
57          super(pos);
58      }
59  
60      public ObjectProperty(int pos, int len) {
61          super(pos, len);
62      }
63  
64      /**
65       * Marks this node as a "getter" property.
66       */
67      public void setIsGetter() {
68          type = Token.GET;
69      }
70  
71      /**
72       * Returns true if this is a getter function.
73       */
74      public boolean isGetter() {
75          return type == Token.GET;
76      }
77  
78      /**
79       * Marks this node as a "setter" property.
80       */
81      public void setIsSetter() {
82          type = Token.SET;
83      }
84  
85      /**
86       * Returns true if this is a setter function.
87       */
88      public boolean isSetter() {
89          return type == Token.SET;
90      }
91  
92      @Override
93      public String toSource(int depth) {
94          StringBuilder sb = new StringBuilder();
95          sb.append(makeIndent(depth));
96          if (isGetter()) {
97              sb.append("get ");
98          } else if (isSetter()) {
99              sb.append("set ");
100         }
101         sb.append(left.toSource(0));
102         if (type == Token.COLON) {
103             sb.append(": ");
104         }
105         sb.append(right.toSource(0));
106         return sb.toString();
107     }

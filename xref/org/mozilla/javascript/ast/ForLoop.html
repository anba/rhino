102     @Override
103     public String toSource(int depth) {
104         StringBuilder sb = new StringBuilder();
105         sb.append(makeIndent(depth));
106         sb.append("for (");
107         sb.append(initializer.toSource(0));
108         sb.append("; ");
109         sb.append(condition.toSource(0));
110         sb.append("; ");
111         sb.append(increment.toSource(0));
112         sb.append(") ");
113         if (body instanceof Block) {
114             sb.append(body.toSource(depth).trim()).append("\n");
115         } else {
116             sb.append("\n").append(body.toSource(depth+1));
117         }
118         return sb.toString();
119     }

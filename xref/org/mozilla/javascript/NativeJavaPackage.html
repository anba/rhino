90      // set up a name which is known to be a package so we don't
91      // need to look for a class by that name
92      NativeJavaPackage forcePackage(String name, Scriptable scope)
93      {
94          Object cached = super.get(name, this);
95          if (cached != null && cached instanceof NativeJavaPackage) {
96              return (NativeJavaPackage) cached;
97          } else {
98              String newPackage = packageName.length() == 0
99                                  ? name
100                                 : packageName + "." + name;
101             NativeJavaPackage pkg = new NativeJavaPackage(true, newPackage, classLoader);
102             ScriptRuntime.setObjectProtoAndParent(pkg, scope);
103             super.put(name, this, pkg);
104             return pkg;
105         }
106     }

108     synchronized Object getPkgProperty(String name, Scriptable start,
109                                        boolean createPkg)
110     {
111         Object cached = super.get(name, start);
112         if (cached != NOT_FOUND)
113             return cached;
114         if (negativeCache != null && negativeCache.contains(name)) {
115             // Performance optimization: see bug 421071
116             return null;
117         }
118 
119         String className = (packageName.length() == 0)
120                                ? name : packageName + '.' + name;
121         Context cx = Context.getContext();
122         ClassShutter shutter = cx.getClassShutter();
123         Scriptable newValue = null;
124         if (shutter == null || shutter.visibleToScripts(className)) {
125             Class<?> cl = null;
126             if (classLoader != null) {
127                 cl = Kit.classOrNull(classLoader, className);
128             } else {
129                 cl = Kit.classOrNull(className);
130             }
131             if (cl != null) {
132                 WrapFactory wrapFactory = cx.getWrapFactory();
133                 newValue = wrapFactory.wrapJavaClass(cx, getTopLevelScope(this), cl);
134                 newValue.setPrototype(getPrototype());
135             }
136         }
137         if (newValue == null) {
138             if (createPkg) {
139                 NativeJavaPackage pkg;
140                 pkg = new NativeJavaPackage(true, className, classLoader);
141                 ScriptRuntime.setObjectProtoAndParent(pkg, getParentScope());
142                 newValue = pkg;
143             } else {
144                 // add to negative cache
145                 if (negativeCache == null)
146                     negativeCache = new HashSet<String>();
147                 negativeCache.add(name);
148             }
149         }
150         if (newValue != null) {
151             // Make it available for fast lookup and sharing of
152             // lazily-reflected constructors and static members.
153             super.put(name, start, newValue);
154         }
155         return newValue;
156     }

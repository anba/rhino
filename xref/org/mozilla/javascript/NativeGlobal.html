48                case Id_escape:

50                  break;

53                  break;

135     public Object execIdCall(IdFunctionObject f, Context cx, Scriptable scope,
136                              Scriptable thisObj, Object[] args)
137     {
138         if (f.hasTag(FTAG)) {
139             int methodId = f.methodId();
140             switch (methodId) {
141                 case Id_decodeURI:
142                 case Id_decodeURIComponent: {
143                     String str = ScriptRuntime.toString(args, 0);
144                     return decode(str, methodId == Id_decodeURI);
145                 }
146 
147                 case Id_encodeURI:
148                 case Id_encodeURIComponent: {
149                     String str = ScriptRuntime.toString(args, 0);
150                     return encode(str, methodId == Id_encodeURI);
151                 }
152 
153                 case Id_escape:
154                     return js_escape(args);
155 
156                 case Id_eval:
157                     return js_eval(cx, scope, args);
158 
159                 case Id_isFinite: {
160                     boolean result;
161                     if (args.length < 1) {
162                         result = false;
163                     } else {
164                         double d = ScriptRuntime.toNumber(args[0]);
165                         result = (d == d
166                                   && d != Double.POSITIVE_INFINITY
167                                   && d != Double.NEGATIVE_INFINITY);
168                     }
169                     return ScriptRuntime.wrapBoolean(result);
170                 }
171 
172                 case Id_isNaN: {
173                     // The global method isNaN, as per ECMA-262 15.1.2.6.
174                     boolean result;
175                     if (args.length < 1) {
176                         result = true;
177                     } else {
178                         double d = ScriptRuntime.toNumber(args[0]);
179                         result = (d != d);
180                     }
181                     return ScriptRuntime.wrapBoolean(result);
182                 }
183 
184                 case Id_isXMLName: {
185                     Object name = (args.length == 0)
186                                   ? Undefined.instance : args[0];
187                     XMLLib xmlLib = XMLLib.extractFromScope(scope);
188                     return ScriptRuntime.wrapBoolean(
189                         xmlLib.isXMLName(cx, name));
190                 }
191 
192                 case Id_parseFloat:
193                     return js_parseFloat(args);
194 
195                 case Id_parseInt:
196                     return js_parseInt(args);
197 
198                 case Id_unescape:
199                     return js_unescape(args);
200 
201                 case Id_uneval: {
202                     Object value = (args.length != 0)
203                                    ? args[0] : Undefined.instance;
204                     return ScriptRuntime.uneval(cx, scope, value);
205                 }
206 
207                 case Id_new_CommonError:
208                     // The implementation of all the ECMA error constructors
209                     // (SyntaxError, TypeError, etc.)
210                     return NativeError.make(cx, scope, f, args);
211             }
212         }
213         throw f.unknown();
214     }

217      * The global method parseInt, as per ECMA-262 15.1.2.2.
218      */
219     private Object js_parseInt(Object[] args) {
220         String s = ScriptRuntime.toString(args, 0);
221         int radix = ScriptRuntime.toInt32(args, 1);
222 
223         int len = s.length();
224         if (len == 0)
225             return ScriptRuntime.NaNobj;
226 
227         boolean negative = false;
228         int start = 0;
229         char c;
230         do {
231             c = s.charAt(start);
232             if (!ScriptRuntime.isStrWhiteSpaceChar(c))
233                 break;
234             start++;
235         } while (start < len);
236 
237         if (c == '+' || (negative = (c == '-')))
238             start++;
239 
240         final int NO_RADIX = -1;
241         if (radix == 0) {
242             radix = NO_RADIX;
243         } else if (radix < 2 || radix > 36) {
244             return ScriptRuntime.NaNobj;
245         } else if (radix == 16 && len - start > 1 && s.charAt(start) == '0') {
246             c = s.charAt(start+1);
247             if (c == 'x' || c == 'X')
248                 start += 2;
249         }
250 
251         if (radix == NO_RADIX) {
252             radix = 10;
253             if (len - start > 1 && s.charAt(start) == '0') {
254                 c = s.charAt(start+1);
255                 if (c == 'x' || c == 'X') {
256                     radix = 16;
257                     start += 2;
258                 } else if ('0' <= c && c <= '9') {
259                     radix = 8;
260                     start++;
261                 }
262             }
263         }
264 
265         double d = ScriptRuntime.stringToNumber(s, start, radix);
266         return ScriptRuntime.wrapNumber(negative ? -d : d);
267     }

269     /**
270      * The global method parseFloat, as per ECMA-262 15.1.2.3.
271      *
272      * @param args the arguments to parseFloat, ignoring args[>=1]
273      */
274     private Object js_parseFloat(Object[] args)
275     {
276         if (args.length < 1)
277             return ScriptRuntime.NaNobj;
278 
279         String s = ScriptRuntime.toString(args[0]);
280         int len = s.length();
281         int start = 0;
282         // Scan forward to skip whitespace
283         char c;
284         for (;;) {
285             if (start == len) {
286                 return ScriptRuntime.NaNobj;
287             }
288             c = s.charAt(start);
289             if (!ScriptRuntime.isStrWhiteSpaceChar(c)) {
290                 break;
291             }
292             ++start;
293         }
294 
295         int i = start;
296         if (c == '+' || c == '-') {
297             ++i;
298             if (i == len) {
299                 return ScriptRuntime.NaNobj;
300             }
301             c = s.charAt(i);
302         }
303 
304         if (c == 'I') {
305             // check for "Infinity"
306             if (i+8 <= len && s.regionMatches(i, "Infinity", 0, 8)) {
307                 double d;
308                 if (s.charAt(start) == '-') {
309                     d = Double.NEGATIVE_INFINITY;
310                 } else {
311                     d = Double.POSITIVE_INFINITY;
312                 }
313                 return ScriptRuntime.wrapNumber(d);
314             }
315             return ScriptRuntime.NaNobj;
316         }
317 
318         // Find the end of the legal bit
319         int decimal = -1;
320         int exponent = -1;
321         boolean exponentValid = false;
322         for (; i < len; i++) {
323             switch (s.charAt(i)) {
324               case '.':
325                 if (decimal != -1) // Only allow a single decimal point.
326                     break;
327                 decimal = i;
328                 continue;
329 
330               case 'e':
331               case 'E':
332                 if (exponent != -1) {
333                     break;
334                 } else if (i == len - 1) {
335                     break;
336                 }
337                 exponent = i;
338                 continue;
339 
340               case '+':
341               case '-':
342                  // Only allow '+' or '-' after 'e' or 'E'
343                 if (exponent != i-1) {
344                     break;
345                 } else if (i == len - 1) {
346                     --i;
347                     break;
348                 }
349                 continue;
350 
351               case '0': case '1': case '2': case '3': case '4':
352               case '5': case '6': case '7': case '8': case '9':
353                 if (exponent != -1) {
354                     exponentValid = true;
355                 }
356                 continue;
357 
358               default:
359                 break;
360             }
361             break;
362         }
363         if (exponent != -1 && !exponentValid) {
364             i = exponent;
365         }
366         s = s.substring(start, i);
367         try {
368             return Double.valueOf(s);
369         }
370         catch (NumberFormatException ex) {
371             return ScriptRuntime.NaNobj;
372         }
373     }

375     /**
376      * The global method escape, as per ECMA-262 15.1.2.4.
377 
378      * Includes code for the 'mask' argument supported by the C escape
379      * method, which used to be part of the browser imbedding.  Blame
380      * for the strange constant names should be directed there.
381      */
382 
383     private Object js_escape(Object[] args) {
384         final int
385             URL_XALPHAS = 1,
386             URL_XPALPHAS = 2,
387             URL_PATH = 4;
388 
389         String s = ScriptRuntime.toString(args, 0);
390 
391         int mask = URL_XALPHAS | URL_XPALPHAS | URL_PATH;
392         if (args.length > 1) { // the 'mask' argument.  Non-ECMA.
393             double d = ScriptRuntime.toNumber(args[1]);
394             if (d != d || ((mask = (int) d) != d) ||
395                 0 != (mask & ~(URL_XALPHAS | URL_XPALPHAS | URL_PATH)))
396             {
397                 throw Context.reportRuntimeError0("msg.bad.esc.mask");
398             }
399         }
400 
401         StringBuffer sb = null;
402         for (int k = 0, L = s.length(); k != L; ++k) {
403             int c = s.charAt(k);
404             if (mask != 0
405                 && ((c >= '0' && c <= '9')
406                     || (c >= 'A' && c <= 'Z') || (c >= 'a' && c <= 'z')
407                     || c == '@' || c == '*' || c == '_' || c == '-' || c == '.'
408                     || (0 != (mask & URL_PATH) && (c == '/' || c == '+'))))
409             {
410                 if (sb != null) {
411                     sb.append((char)c);
412                 }
413             } else {
414                 if (sb == null) {
415                     sb = new StringBuffer(L + 3);
416                     sb.append(s);
417                     sb.setLength(k);
418                 }
419 
420                 int hexSize;
421                 if (c < 256) {
422                     if (c == ' ' && mask == URL_XPALPHAS) {
423                         sb.append('+');
424                         continue;
425                     }
426                     sb.append('%');
427                     hexSize = 2;
428                 } else {
429                     sb.append('%');
430                     sb.append('u');
431                     hexSize = 4;
432                 }
433 
434                 // append hexadecimal form of c left-padded with 0
435                 for (int shift = (hexSize - 1) * 4; shift >= 0; shift -= 4) {
436                     int digit = 0xf & (c >> shift);
437                     int hc = (digit < 10) ? '0' + digit : 'A' - 10 + digit;
438                     sb.append((char)hc);
439                 }
440             }
441         }
442 
443         return (sb == null) ? s : sb.toString();
444     }

450     private Object js_unescape(Object[] args)
451     {
452         String s = ScriptRuntime.toString(args, 0);
453         int firstEscapePos = s.indexOf('%');
454         if (firstEscapePos >= 0) {
455             int L = s.length();
456             char[] buf = s.toCharArray();
457             int destination = firstEscapePos;
458             for (int k = firstEscapePos; k != L;) {
459                 char c = buf[k];
460                 ++k;
461                 if (c == '%' && k != L) {
462                     int end, start;
463                     if (buf[k] == 'u') {
464                         start = k + 1;
465                         end = k + 5;
466                     } else {
467                         start = k;
468                         end = k + 2;
469                     }
470                     if (end <= L) {
471                         int x = 0;
472                         for (int i = start; i != end; ++i) {
473                             x = Kit.xDigitToInt(buf[i], x);
474                         }
475                         if (x >= 0) {
476                             c = (char)x;
477                             k = end;
478                         }
479                     }
480                 }
481                 buf[destination] = c;
482                 ++destination;
483             }
484             s = new String(buf, 0, destination);
485         }
486         return s;
487     }

541     /*
542     *   ECMA 3, 15.1.3 URI Handling Function Properties
543     *
544     *   The following are implementations of the algorithms
545     *   given in the ECMA specification for the hidden functions
546     *   'Encode' and 'Decode'.
547     */
548     private static String encode(String str, boolean fullUri) {
549         byte[] utf8buf = null;
550         StringBuffer sb = null;
551 
552         for (int k = 0, length = str.length(); k != length; ++k) {
553             char C = str.charAt(k);
554             if (encodeUnescaped(C, fullUri)) {
555                 if (sb != null) {
556                     sb.append(C);
557                 }
558             } else {
559                 if (sb == null) {
560                     sb = new StringBuffer(length + 3);
561                     sb.append(str);
562                     sb.setLength(k);
563                     utf8buf = new byte[6];
564                 }
565                 if (0xDC00 <= C && C <= 0xDFFF) {
566                     throw uriError();
567                 }
568                 int V;
569                 if (C < 0xD800 || 0xDBFF < C) {
570                     V = C;
571                 } else {
572                     k++;
573                     if (k == length) {
574                         throw uriError();
575                     }
576                     char C2 = str.charAt(k);
577                     if (!(0xDC00 <= C2 && C2 <= 0xDFFF)) {
578                         throw uriError();
579                     }
580                     V = ((C - 0xD800) << 10) + (C2 - 0xDC00) + 0x10000;
581                 }
582                 int L = oneUcs4ToUtf8Char(utf8buf, V);
583                 for (int j = 0; j < L; j++) {
584                     int d = 0xff & utf8buf[j];
585                     sb.append('%');
586                     sb.append(toHexChar(d >>> 4));
587                     sb.append(toHexChar(d & 0xf));
588                 }
589             }
590         }
591         return (sb == null) ? str : sb.toString();
592     }

594     private static char toHexChar(int i) {
595         if (i >> 4 != 0) Kit.codeBug();
596         return (char)((i < 10) ? i + '0' : i - 10 + 'A');
597     }

599     private static int unHex(char c) {
600         if ('A' <= c && c <= 'F') {
601             return c - 'A' + 10;
602         } else if ('a' <= c && c <= 'f') {
603             return c - 'a' + 10;
604         } else if ('0' <= c && c <= '9') {
605             return c - '0';
606         } else {
607             return -1;
608         }
609     }

611     private static int unHex(char c1, char c2) {
612         int i1 = unHex(c1);
613         int i2 = unHex(c2);
614         if (i1 >= 0 && i2 >= 0) {
615             return (i1 << 4) | i2;
616         }
617         return -1;
618     }

620     private static String decode(String str, boolean fullUri) {
621         char[] buf = null;
622         int bufTop = 0;
623 
624         for (int k = 0, length = str.length(); k != length;) {
625             char C = str.charAt(k);
626             if (C != '%') {
627                 if (buf != null) {
628                     buf[bufTop++] = C;
629                 }
630                 ++k;
631             } else {
632                 if (buf == null) {
633                     // decode always compress so result can not be bigger then
634                     // str.length()
635                     buf = new char[length];
636                     str.getChars(0, k, buf, 0);
637                     bufTop = k;
638                 }
639                 int start = k;
640                 if (k + 3 > length)
641                     throw uriError();
642                 int B = unHex(str.charAt(k + 1), str.charAt(k + 2));
643                 if (B < 0) throw uriError();
644                 k += 3;
645                 if ((B & 0x80) == 0) {
646                     C = (char)B;
647                 } else {
648                     // Decode UTF-8 sequence into ucs4Char and encode it into
649                     // UTF-16
650                     int utf8Tail, ucs4Char, minUcs4Char;
651                     if ((B & 0xC0) == 0x80) {
652                         // First  UTF-8 should be ouside 0x80..0xBF
653                         throw uriError();
654                     } else if ((B & 0x20) == 0) {
655                         utf8Tail = 1; ucs4Char = B & 0x1F;
656                         minUcs4Char = 0x80;
657                     } else if ((B & 0x10) == 0) {
658                         utf8Tail = 2; ucs4Char = B & 0x0F;
659                         minUcs4Char = 0x800;
660                     } else if ((B & 0x08) == 0) {
661                         utf8Tail = 3; ucs4Char = B & 0x07;
662                         minUcs4Char = 0x10000;
663                     } else if ((B & 0x04) == 0) {
664                         utf8Tail = 4; ucs4Char = B & 0x03;
665                         minUcs4Char = 0x200000;
666                     } else if ((B & 0x02) == 0) {
667                         utf8Tail = 5; ucs4Char = B & 0x01;
668                         minUcs4Char = 0x4000000;
669                     } else {
670                         // First UTF-8 can not be 0xFF or 0xFE
671                         throw uriError();
672                     }
673                     if (k + 3 * utf8Tail > length)
674                         throw uriError();
675                     for (int j = 0; j != utf8Tail; j++) {
676                         if (str.charAt(k) != '%')
677                             throw uriError();
678                         B = unHex(str.charAt(k + 1), str.charAt(k + 2));
679                         if (B < 0 || (B & 0xC0) != 0x80)
680                             throw uriError();
681                         ucs4Char = (ucs4Char << 6) | (B & 0x3F);
682                         k += 3;
683                     }
684                     // Check for overlongs and other should-not-present codes
685                     if (ucs4Char < minUcs4Char
686                             || (ucs4Char >= 0xD800 && ucs4Char <= 0xDFFF)) {
687                         ucs4Char = INVALID_UTF8;
688                     } else if (ucs4Char == 0xFFFE || ucs4Char == 0xFFFF) {
689                         ucs4Char = 0xFFFD;
690                     }
691                     if (ucs4Char >= 0x10000) {
692                         ucs4Char -= 0x10000;
693                         if (ucs4Char > 0xFFFFF) {
694                             throw uriError();
695                         }
696                         char H = (char)((ucs4Char >>> 10) + 0xD800);
697                         C = (char)((ucs4Char & 0x3FF) + 0xDC00);
698                         buf[bufTop++] = H;
699                     } else {
700                         C = (char)ucs4Char;
701                     }
702                 }
703                 if (fullUri && URI_DECODE_RESERVED.indexOf(C) >= 0) {
704                     for (int x = start; x != k; x++) {
705                         buf[bufTop++] = str.charAt(x);
706                     }
707                 } else {
708                     buf[bufTop++] = C;
709                 }
710             }
711         }
712         return (buf == null) ? str : new String(buf, 0, bufTop);
713     }

715     private static boolean encodeUnescaped(char c, boolean fullUri) {
716         if (('A' <= c && c <= 'Z') || ('a' <= c && c <= 'z')
717                 || ('0' <= c && c <= '9')) {
718             return true;
719         }
720         if ("-_.!~*'()".indexOf(c) >= 0) {
721             return true;
722         }
723         if (fullUri) {
724             return URI_DECODE_RESERVED.indexOf(c) >= 0;
725         }
726         return false;
727     }

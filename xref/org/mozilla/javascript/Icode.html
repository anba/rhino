226     static boolean validIcode(int icode)
227     {
228         return MIN_ICODE <= icode && icode <= 0;
229     }
230 
231     static boolean validTokenCode(int token)
232     {
233         return Token.FIRST_BYTECODE_TOKEN <= token
234                && token <= Token.LAST_BYTECODE_TOKEN;
235     }
236 
237     static boolean validBytecode(int bytecode)
238     {
239         return validIcode(bytecode) || validTokenCode(bytecode);
240     }

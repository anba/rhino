87      @Override
88      public Object execIdCall(IdFunctionObject f, Context cx, Scriptable scope,
89                               Scriptable thisObj, Object[] args)
90      {
91          if (!f.hasTag(MATH_TAG)) {
92              return super.execIdCall(f, cx, scope, thisObj, args);
93          }
94          double x;
95          int methodId = f.methodId();
96          switch (methodId) {
97              case Id_toSource:
98                  return "Math";
99  
100             case Id_abs:
101                 x = ScriptRuntime.toNumber(args, 0);
102                 // abs(-0.0) should be 0.0, but -0.0 < 0.0 == false
103                 x = (x == 0.0) ? 0.0 : (x < 0.0) ? -x : x;
104                 break;
105 
106             case Id_acos:
107             case Id_asin:
108                 x = ScriptRuntime.toNumber(args, 0);
109                 if (x == x && -1.0 <= x && x <= 1.0) {
110                     x = (methodId == Id_acos) ? Math.acos(x) : Math.asin(x);
111                 } else {
112                     x = Double.NaN;
113                 }
114                 break;
115 
116             case Id_atan:
117                 x = ScriptRuntime.toNumber(args, 0);
118                 x = Math.atan(x);
119                 break;
120 
121             case Id_atan2:
122                 x = ScriptRuntime.toNumber(args, 0);
123                 x = Math.atan2(x, ScriptRuntime.toNumber(args, 1));
124                 break;
125 
126             case Id_ceil:
127                 x = ScriptRuntime.toNumber(args, 0);
128                 x = Math.ceil(x);
129                 break;
130 
131             case Id_cos:
132                 x = ScriptRuntime.toNumber(args, 0);
133                 x = (x == Double.POSITIVE_INFINITY
134                      || x == Double.NEGATIVE_INFINITY)
135                     ? Double.NaN : Math.cos(x);
136                 break;
137 
138             case Id_exp:
139                 x = ScriptRuntime.toNumber(args, 0);
140                 x = (x == Double.POSITIVE_INFINITY) ? x
141                     : (x == Double.NEGATIVE_INFINITY) ? 0.0
142                     : Math.exp(x);
143                 break;
144 
145             case Id_floor:
146                 x = ScriptRuntime.toNumber(args, 0);
147                 x = Math.floor(x);
148                 break;
149 
150             case Id_log:
151                 x = ScriptRuntime.toNumber(args, 0);
152                 // Java's log(<0) = -Infinity; we need NaN
153                 x = (x < 0) ? Double.NaN : Math.log(x);
154                 break;
155 
156             case Id_max:
157             case Id_min:
158                 x = (methodId == Id_max)
159                     ? Double.NEGATIVE_INFINITY : Double.POSITIVE_INFINITY;
160                 for (int i = 0; i != args.length; ++i) {
161                     double d = ScriptRuntime.toNumber(args[i]);
162                     if (d != d) {
163                         x = d; // NaN
164                         break;
165                     }
166                     if (methodId == Id_max) {
167                         // if (x < d) x = d; does not work due to -0.0 >= +0.0
168                         x = Math.max(x, d);
169                     } else {
170                         x = Math.min(x, d);
171                     }
172                 }
173                 break;
174 
175             case Id_pow:
176                 x = ScriptRuntime.toNumber(args, 0);
177                 x = js_pow(x, ScriptRuntime.toNumber(args, 1));
178                 break;
179 
180             case Id_random:
181                 x = Math.random();
182                 break;
183 
184             case Id_round:
185                 x = ScriptRuntime.toNumber(args, 0);
186                 if (x == x && x != Double.POSITIVE_INFINITY
187                     && x != Double.NEGATIVE_INFINITY)
188                 {
189                     // Round only finite x
190                     long l = Math.round(x);
191                     if (l != 0) {
192                         x = l;
193                     } else {
194                         // We must propagate the sign of d into the result
195                         if (x < 0.0) {
196                             x = ScriptRuntime.negativeZero;
197                         } else if (x != 0.0) {
198                             x = 0.0;
199                         }
200                     }
201                 }
202                 break;
203 
204             case Id_sin:
205                 x = ScriptRuntime.toNumber(args, 0);
206                 x = (x == Double.POSITIVE_INFINITY
207                      || x == Double.NEGATIVE_INFINITY)
208                     ? Double.NaN : Math.sin(x);
209                 break;
210 
211             case Id_sqrt:
212                 x = ScriptRuntime.toNumber(args, 0);
213                 x = Math.sqrt(x);
214                 break;
215 
216             case Id_tan:
217                 x = ScriptRuntime.toNumber(args, 0);
218                 x = Math.tan(x);
219                 break;
220 
221             default: throw new IllegalStateException(String.valueOf(methodId));
222         }
223         return ScriptRuntime.wrapNumber(x);
224     }

226     // See Ecma 15.8.2.13
227     private double js_pow(double x, double y) {
228         double result;
229         if (y != y) {
230             // y is NaN, result is always NaN
231             result = y;
232         } else if (y == 0) {
233             // Java's pow(NaN, 0) = NaN; we need 1
234             result = 1.0;
235         } else if (x == 0) {
236             // Many differences from Java's Math.pow
237             if (1 / x > 0) {
238                 result = (y > 0) ? 0 : Double.POSITIVE_INFINITY;
239             } else {
240                 // x is -0, need to check if y is an odd integer
241                 long y_long = (long)y;
242                 if (y_long == y && (y_long & 0x1) != 0) {
243                     result = (y > 0) ? -0.0 : Double.NEGATIVE_INFINITY;
244                 } else {
245                     result = (y > 0) ? 0.0 : Double.POSITIVE_INFINITY;
246                 }
247             }
248         } else {
249             result = Math.pow(x, y);
250             if (result != result) {
251                 // Check for broken Java implementations that gives NaN
252                 // when they should return something else
253                 if (y == Double.POSITIVE_INFINITY) {
254                     if (x < -1.0 || 1.0 < x) {
255                         result = Double.POSITIVE_INFINITY;
256                     } else if (-1.0 < x && x < 1.0) {
257                         result = 0;
258                     }
259                 } else if (y == Double.NEGATIVE_INFINITY) {
260                     if (x < -1.0 || 1.0 < x) {
261                         result = 0;
262                     } else if (-1.0 < x && x < 1.0) {
263                         result = Double.POSITIVE_INFINITY;
264                     }
265                 } else if (x == Double.POSITIVE_INFINITY) {
266                     result = (y > 0) ? Double.POSITIVE_INFINITY : 0.0;
267                 } else if (x == Double.NEGATIVE_INFINITY) {
268                     long y_long = (long)y;
269                     if (y_long == y && (y_long & 0x1) != 0) {
270                         // y is odd integer
271                         result = (y > 0) ? Double.NEGATIVE_INFINITY : -0.0;
272                     } else {
273                         result = (y > 0) ? Double.POSITIVE_INFINITY : 0.0;
274                     }
275                 }
276             }
277         }
278         return result;
279     }

51      public static String getMessage(String messageId, Object[] args) {
52          Context cx = Context.getCurrentContext();
53          Locale locale = cx == null ? Locale.getDefault() : cx.getLocale();
54  
55          // ResourceBundle does caching.
56          ResourceBundle rb = ResourceBundle.getBundle
57              ("org.mozilla.javascript.tools.resources.Messages", locale);
58  
59          String formatString;
60          try {
61              formatString = rb.getString(messageId);
62          } catch (java.util.MissingResourceException mre) {
63              throw new RuntimeException("no message resource found for message property "
64                                         + messageId);
65          }
66  
67          if (args == null) {
68              return formatString;
69          } else {
70              MessageFormat formatter = new MessageFormat(formatString);
71              return formatter.format(args);
72          }
73      }

75      private static String getExceptionMessage(RhinoException ex)
76      {
77          String msg;
78          if (ex instanceof JavaScriptException) {
79              msg = getMessage("msg.uncaughtJSException", ex.details());
80          } else if (ex instanceof EcmaError) {
81              msg = getMessage("msg.uncaughtEcmaError", ex.details());
82          } else if (ex instanceof EvaluatorException) {
83              msg = ex.details();
84          } else {
85              msg = ex.toString();
86          }
87          return msg;
88      }

127     public static void reportException(ErrorReporter er, RhinoException ex)
128     {
129         if (er instanceof ToolErrorReporter) {
130             ((ToolErrorReporter)er).reportException(ex);
131         } else {
132             String msg = getExceptionMessage(ex);
133             er.error(msg, ex.sourceName(), ex.lineNumber(),
134                      ex.lineSource(), ex.columnNumber());
135         }
136     }

138     public void reportException(RhinoException ex)
139     {
140         if (ex instanceof WrappedException) {
141             WrappedException we = (WrappedException)ex;
142             we.printStackTrace(err);
143         } else {
144             String lineSeparator =
145                 SecurityUtilities.getSystemProperty("line.separator");
146             String msg = getExceptionMessage(ex) + lineSeparator +
147                 ex.getScriptStackTrace();
148             reportErrorMessage(msg, ex.sourceName(), ex.lineNumber(),
149                                ex.lineSource(), ex.columnNumber(), false);
150         }
151     }

153     private void reportErrorMessage(String message, String sourceName, int line,
154                                     String lineSource, int lineOffset,
155                                     boolean justWarning)
156     {
157         if (line > 0) {
158             String lineStr = String.valueOf(line);
159             if (sourceName != null) {
160                 Object[] args = { sourceName, lineStr, message };
161                 message = getMessage("msg.format3", args);
162             } else {
163                 Object[] args = { lineStr, message };
164                 message = getMessage("msg.format2", args);
165             }
166         } else {
167             Object[] args = { message };
168             message = getMessage("msg.format1", args);
169         }
170         if (justWarning) {
171             message = getMessage("msg.warning", message);
172         }
173         err.println(messagePrefix + message);
174         if (null != lineSource) {
175             err.println(messagePrefix + lineSource);
176             err.println(messagePrefix + buildIndicator(lineOffset));
177         }
178     }
179 
180     private String buildIndicator(int offset){
181         StringBuffer sb = new StringBuffer();
182         for (int i = 0; i < offset-1; i++)
183             sb.append(".");
184         sb.append("^");
185         return sb.toString();
186     }

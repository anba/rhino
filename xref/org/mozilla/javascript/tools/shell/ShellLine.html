21  /**
22   * Provides a specialized input stream for consoles to handle line
23   * editing, history and completion. Relies on the JLine library (see
24   * <http://jline.sourceforge.net>).
25   */
26  public class ShellLine {
27  
28      public static InputStream getStream(Scriptable scope) {
29          // We don't want a compile-time dependency on the JLine jar, so use
30          // reflection to load and reference the JLine classes.
31          ClassLoader classLoader = ShellLine.class.getClassLoader();
32          if (classLoader == null) {
33              // If the attempt to get a class specific class loader above failed
34              // then fallback to the system class loader.
35              classLoader = ClassLoader.getSystemClassLoader();
36          }
37          if (classLoader == null) {
38              // If for some reason we still don't have a handle to a class
39              // loader then give up (avoid a NullPointerException).
40              return null;
41          }
42          Class<?> readerClass = Kit.classOrNull(classLoader, "jline.ConsoleReader");
43          if (readerClass == null)
44              return null;
45          try {
46              // ConsoleReader reader = new ConsoleReader();
47              Constructor<?> c = readerClass.getConstructor();
48              Object reader = c.newInstance();
49  
50              // reader.setBellEnabled(false);
51              Method m = readerClass.getMethod("setBellEnabled", Boolean.TYPE);
52              m.invoke(reader, Boolean.FALSE);
53  
54              // reader.addCompletor(new FlexibleCompletor(prefixes));
55              Class<?> completorClass = Kit.classOrNull(classLoader,
56                  "jline.Completor");
57              m = readerClass.getMethod("addCompletor", completorClass);
58              Object completor = Proxy.newProxyInstance(classLoader,
59                      new Class[] { completorClass },
60                      new FlexibleCompletor(completorClass, scope));
61              m.invoke(reader, completor);
62  
63              // return new ConsoleReaderInputStream(reader);
64              Class<?> inputStreamClass = Kit.classOrNull(classLoader,
65                  "jline.ConsoleReaderInputStream");
66              c = inputStreamClass.getConstructor(readerClass);
67              return (InputStream) c.newInstance(reader);
68          } catch (NoSuchMethodException e) {
69          } catch (InstantiationException e) {
70          } catch (IllegalAccessException e) {
71          } catch (InvocationTargetException e) {
72          }
73          return null;
74      }
75  }

77  /**
78   * The completors provided with JLine are pretty uptight, they only
79   * complete on a line that it can fully recognize (only composed of
80   * completed strings). This one completes whatever came before.
81   */
82  class FlexibleCompletor implements java.lang.reflect.InvocationHandler {
83      private Method completeMethod;
84      private Scriptable global;
85  
86      FlexibleCompletor(Class<?> completorClass, Scriptable global)
87          throws NoSuchMethodException
88      {
89          this.global = global;
90          this.completeMethod = completorClass.getMethod("complete", String.class,
91                  Integer.TYPE, List.class);
92      }
93  
94      @SuppressWarnings({"unchecked"})
95      public Object invoke(Object proxy, Method method, Object[] args) {
96          if (method.equals(this.completeMethod)) {
97              int result = complete((String)args[0], ((Integer) args[1]).intValue(),
98                      (List<String>) args[2]);
99              return Integer.valueOf(result);
100         }
101         throw new NoSuchMethodError(method.toString());
102     }
103 
104     public int complete(String buffer, int cursor, List<String> candidates) {
105         // Starting from "cursor" at the end of the buffer, look backward
106         // and collect a list of identifiers separated by (possibly zero)
107         // dots. Then look up each identifier in turn until getting to the
108         // last, presumably incomplete fragment. Then enumerate all the
109         // properties of the last object and find any that have the
110         // fragment as a prefix and return those for autocompletion.
111         int m = cursor - 1;
112         while (m >= 0) {
113             char c = buffer.charAt(m);
114             if (!Character.isJavaIdentifierPart(c) && c != '.')
115                 break;
116             m--;
117         }
118         String namesAndDots = buffer.substring(m+1, cursor);
119         String[] names = namesAndDots.split("\\.", -1);
120         Scriptable obj = this.global;
121         for (int i=0; i < names.length - 1; i++) {
122             Object val = obj.get(names[i], global);
123             if (val instanceof Scriptable)
124                 obj = (Scriptable) val;
125             else {
126                 return buffer.length(); // no matches
127             }
128         }
129         Object[] ids = (obj instanceof ScriptableObject)
130                        ? ((ScriptableObject)obj).getAllIds()
131                        : obj.getIds();
132         String lastPart = names[names.length-1];
133         for (int i=0; i < ids.length; i++) {
134             if (!(ids[i] instanceof String))
135                 continue;
136             String id = (String)ids[i];
137             if (id.startsWith(lastPart)) {
138                 if (obj.get(id, obj) instanceof Function)
139                     id += "(";
140                 candidates.add(id);
141             }
142         }
143         return buffer.length() - lastPart.length();
144     }

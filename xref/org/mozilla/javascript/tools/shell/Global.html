125     public Require installRequire(Context cx, List<String> modulePath,
126                                   boolean sandboxed) {
127         RequireBuilder rb = new RequireBuilder();
128         rb.setSandboxed(sandboxed);
129         List<URI> uris = new ArrayList<URI>();
130         if (modulePath != null) {
131             for (String path : modulePath) {
132                 try {
133                     URI uri = new URI(path);
134                     if (!uri.isAbsolute()) {
135                         // call resolve("") to canonify the path
136                         uri = new File(path).toURI().resolve("");
137                     }
138                     if (!uri.toString().endsWith("/")) {
139                         // make sure URI always terminates with slash to
140                         // avoid loading from unintended locations
141                         uri = new URI(uri + "/");
142                     }
143                     uris.add(uri);
144                 } catch (URISyntaxException usx) {
145                     throw new RuntimeException(usx);
146                 }
147             }
148         }
149         rb.setModuleScriptProvider(
150                 new SoftCachingModuleScriptProvider(
151                         new UrlModuleSourceProvider(uris, null)));
152         Require require = rb.createRequire(cx, this);
153         require.install(this);
154         return require;
155     }

314     private static Class<?> getClass(Object[] args) {
315         if (args.length == 0) {
316             throw reportRuntimeError("msg.expected.string.arg");
317         }
318         Object arg0 = args[0];
319         if (arg0 instanceof Wrapper) {
320             Object wrapped = ((Wrapper)arg0).unwrap();
321             if (wrapped instanceof Class)
322                 return (Class<?>)wrapped;
323         }
324         String className = Context.toString(args[0]);
325         try {
326             return Class.forName(className);
327         }
328         catch (ClassNotFoundException cnfe) {
329             throw reportRuntimeError("msg.class.not.found", className);
330         }
331     }

369         if (ScriptableObject.hasProperty(this, "prompts")) {
370             Object promptsJS = ScriptableObject.getProperty(this,
371                                                             "prompts");
372             if (promptsJS instanceof Scriptable) {
373                 Scriptable s = (Scriptable) promptsJS;
374                 if (ScriptableObject.hasProperty(s, 0) &&
375                     ScriptableObject.hasProperty(s, 1))
376                 {
377                     Object elem0 = ScriptableObject.getProperty(s, 0);
378                     if (elem0 instanceof Function) {
379                         elem0 = ((Function) elem0).call(cx, this, s,
380                                 new Object[0]);
381                     }
382                     prompts[0] = Context.toString(elem0);
383                     Object elem1 = ScriptableObject.getProperty(s, 1);
384                     if (elem1 instanceof Function) {
385                         elem1 = ((Function) elem1).call(cx, this, s,
386                                 new Object[0]);
387                     }
388                     prompts[1] = Context.toString(elem1);
389                 }
390             }
391         }
392         return prompts;
393     }

410     public int runDoctest(Context cx, Scriptable scope, String session,
411                           String sourceName, int lineNumber)
412     {
413         doctestCanonicalizations = new HashMap<String,String>();
414         String[] lines = session.split("[\n\r]+");
415         String prompt0 = this.prompts[0].trim();
416         String prompt1 = this.prompts[1].trim();
417         int testCount = 0;
418         int i = 0;
419         while (i < lines.length && !lines[i].trim().startsWith(prompt0)) {
420             i++; // skip lines that don't look like shell sessions
421         }
422     	while (i < lines.length) {
423     		String inputString = lines[i].trim().substring(prompt0.length());
424             inputString += "\n";
425     		i++;
426     		while (i < lines.length && lines[i].trim().startsWith(prompt1)) {
427     			inputString += lines[i].trim().substring(prompt1.length());
428     			inputString += "\n";
429     			i++;
430     		}
431             String expectedString = "";
432             while (i < lines.length &&
433                    !lines[i].trim().startsWith(prompt0))
434             {
435                 expectedString += lines[i] + "\n";
436                 i++;
437             }
438     		PrintStream savedOut = this.getOut();
439     		PrintStream savedErr = this.getErr();
440     		ByteArrayOutputStream out = new ByteArrayOutputStream();
441     		ByteArrayOutputStream err = new ByteArrayOutputStream();
442     		this.setOut(new PrintStream(out));
443     		this.setErr(new PrintStream(err));
444     		String resultString = "";
445     		ErrorReporter savedErrorReporter = cx.getErrorReporter();
446     		cx.setErrorReporter(new ToolErrorReporter(false, this.getErr()));
447     		try {
448     		    testCount++;
449 	    		Object result = cx.evaluateString(scope, inputString,
450 	    				            "doctest input", 1, null);
451 	            if (result != Context.getUndefinedValue() &&
452 	                    !(result instanceof Function &&
453 	                      inputString.trim().startsWith("function")))
454 	            {
455 	            	resultString = Context.toString(result);
456 	            }
457     		} catch (RhinoException e) {
458                 ToolErrorReporter.reportException(cx.getErrorReporter(), e);
459     		} finally {
460     		    this.setOut(savedOut);
461     		    this.setErr(savedErr);
462         		cx.setErrorReporter(savedErrorReporter);
463     			resultString += err.toString() + out.toString();
464     		}
465     		if (!doctestOutputMatches(expectedString, resultString)) {
466     		    String message = "doctest failure running:\n" +
467                     inputString +
468                     "expected: " + expectedString +
469                     "actual: " + resultString + "\n";
470     		    if (sourceName != null)
471                     throw Context.reportRuntimeError(message, sourceName,
472                             lineNumber+i-1, null, 0);
473     		    else
474                     throw Context.reportRuntimeError(message);
475     		}
476     	}
477     	return testCount;
478     }

480     /**
481      * Compare actual result of doctest to expected, modulo some
482      * acceptable differences. Currently just trims the strings
483      * before comparing, but should ignore differences in line numbers
484      * for error messages for example.
485      *
486      * @param expected the expected string
487      * @param actual the actual string
488      * @return true iff actual matches expected modulo some acceptable
489      *      differences
490      */
491     private boolean doctestOutputMatches(String expected, String actual) {
492         expected = expected.trim();
493         actual = actual.trim().replace("\r\n", "\n");
494         if (expected.equals(actual))
495             return true;
496         for (Map.Entry<String,String> entry: doctestCanonicalizations.entrySet()) {
497             expected = expected.replace(entry.getKey(), entry.getValue());
498         }
499         if (expected.equals(actual))
500             return true;
501         // java.lang.Object.toString() prints out a unique hex number associated
502         // with each object. This number changes from run to run, so we want to
503         // ignore differences between these numbers in the output. We search for a
504         // regexp that matches the hex number preceded by '@', then enter mappings into
505         // "doctestCanonicalizations" so that we ensure that the mappings are
506         // consistent within a session.
507         Pattern p = Pattern.compile("@[0-9a-fA-F]+");
508         Matcher expectedMatcher = p.matcher(expected);
509         Matcher actualMatcher = p.matcher(actual);
510         for (;;) {
511             if (!expectedMatcher.find())
512                 return false;
513             if (!actualMatcher.find())
514                 return false;
515             if (actualMatcher.start() != expectedMatcher.start())
516                 return false;
517             int start = expectedMatcher.start();
518             if (!expected.substring(0, start).equals(actual.substring(0, start)))
519                 return false;
520             String expectedGroup = expectedMatcher.group();
521             String actualGroup = actualMatcher.group();
522             String mapping = doctestCanonicalizations.get(expectedGroup);
523             if (mapping == null) {
524                 doctestCanonicalizations.put(expectedGroup, actualGroup);
525                 expected = expected.replace(expectedGroup, actualGroup);
526             } else if (!actualGroup.equals(mapping)) {
527                 return false; // wrong object!
528             }
529             if (expected.equals(actual))
530                 return true;
531         }
532     }

534     /**
535      * The spawn function runs a given function or script in a different
536      * thread.
537      *
538      * js> function g() { a = 7; }
539      * js> a = 3;
540      * 3
541      * js> spawn(g)
542      * Thread[Thread-1,5,main]
543      * js> a
544      * 3
545      */
546     public static Object spawn(Context cx, Scriptable thisObj, Object[] args,
547                                Function funObj)
548     {
549         Scriptable scope = funObj.getParentScope();
550         Runner runner;
551         if (args.length != 0 && args[0] instanceof Function) {
552             Object[] newArgs = null;
553             if (args.length > 1 && args[1] instanceof Scriptable) {
554                 newArgs = cx.getElements((Scriptable) args[1]);
555             }
556             if (newArgs == null) { newArgs = ScriptRuntime.emptyArgs; }
557             runner = new Runner(scope, (Function) args[0], newArgs);
558         } else if (args.length != 0 && args[0] instanceof Script) {
559             runner = new Runner(scope, (Script) args[0]);
560         } else {
561             throw reportRuntimeError("msg.spawn.args");
562         }
563         runner.factory = cx.getFactory();
564         Thread thread = new Thread(runner);
565         thread.start();
566         return thread;
567     }

594             Object syncObject = null;

596                 syncObject = args[1];
597             }
598             return new Synchronizer((Function)args[0], syncObject);
599         }
600         else {
601             throw reportRuntimeError("msg.sync.args");

605     /**
606      * Execute the specified command with the given argument and options
607      * as a separate process and return the exit status of the process.
608      * <p>
609      * Usage:
610      * <pre>
611      * runCommand(command)
612      * runCommand(command, arg1, ..., argN)
613      * runCommand(command, arg1, ..., argN, options)
614      * </pre>
615      * All except the last arguments to runCommand are converted to strings
616      * and denote command name and its arguments. If the last argument is a
617      * JavaScript object, it is an option object. Otherwise it is converted to
618      * string denoting the last argument and options objects assumed to be
619      * empty.
620      * The following properties of the option object are processed:
621      * <ul>
622      * <li><tt>args</tt> - provides an array of additional command arguments
623      * <li><tt>env</tt> - explicit environment object. All its enumerable
624      *   properties define the corresponding environment variable names.
625      * <li><tt>input</tt> - the process input. If it is not
626      *   java.io.InputStream, it is converted to string and sent to the process
627      *   as its input. If not specified, no input is provided to the process.
628      * <li><tt>output</tt> - the process output instead of
629      *   java.lang.System.out. If it is not instance of java.io.OutputStream,
630      *   the process output is read, converted to a string, appended to the
631      *   output property value converted to string and put as the new value of
632      *   the output property.
633      * <li><tt>err</tt> - the process error output instead of
634      *   java.lang.System.err. If it is not instance of java.io.OutputStream,
635      *   the process error output is read, converted to a string, appended to
636      *   the err property value converted to string and put as the new
637      *   value of the err property.
638      * </ul>
639      */
640     public static Object runCommand(Context cx, Scriptable thisObj,
641                                     Object[] args, Function funObj)
642         throws IOException
643     {
644         int L = args.length;
645         if (L == 0 || (L == 1 && args[0] instanceof Scriptable)) {
646             throw reportRuntimeError("msg.runCommand.bad.args");
647         }
648 
649         InputStream in = null;
650         OutputStream out = null, err = null;
651         ByteArrayOutputStream outBytes = null, errBytes = null;
652         Object outObj = null, errObj = null;
653         String[] environment = null;
654         Scriptable params = null;
655         Object[] addArgs = null;
656         if (args[L - 1] instanceof Scriptable) {
657             params = (Scriptable)args[L - 1];
658             --L;
659             Object envObj = ScriptableObject.getProperty(params, "env");
660             if (envObj != Scriptable.NOT_FOUND) {
661                 if (envObj == null) {
662                     environment = new String[0];
663                 } else {
664                     if (!(envObj instanceof Scriptable)) {
665                         throw reportRuntimeError("msg.runCommand.bad.env");
666                     }
667                     Scriptable envHash = (Scriptable)envObj;
668                     Object[] ids = ScriptableObject.getPropertyIds(envHash);
669                     environment = new String[ids.length];
670                     for (int i = 0; i != ids.length; ++i) {
671                         Object keyObj = ids[i], val;
672                         String key;
673                         if (keyObj instanceof String) {
674                             key = (String)keyObj;
675                             val = ScriptableObject.getProperty(envHash, key);
676                         } else {
677                             int ikey = ((Number)keyObj).intValue();
678                             key = Integer.toString(ikey);
679                             val = ScriptableObject.getProperty(envHash, ikey);
680                         }
681                         if (val == ScriptableObject.NOT_FOUND) {
682                             val = Undefined.instance;
683                         }
684                         environment[i] = key+'='+ScriptRuntime.toString(val);
685                     }
686                 }
687             }
688             Object inObj = ScriptableObject.getProperty(params, "input");
689             if (inObj != Scriptable.NOT_FOUND) {
690                 in = toInputStream(inObj);
691             }
692             outObj = ScriptableObject.getProperty(params, "output");
693             if (outObj != Scriptable.NOT_FOUND) {
694                 out = toOutputStream(outObj);
695                 if (out == null) {
696                     outBytes = new ByteArrayOutputStream();
697                     out = outBytes;
698                 }
699             }

709             if (addArgsObj != Scriptable.NOT_FOUND) {

711                                                 getTopLevelScope(thisObj));
712                 addArgs = cx.getElements(s);
713             }
714         }

716         if (out == null) {
717             out = (global != null) ? global.getOut() : System.out;
718         }
719         if (err == null) {

721         }
722         // If no explicit input stream, do not send any input to process,
723         // in particular, do not use System.in to avoid deadlocks
724         // when waiting for user input to send to process which is already
725         // terminated as it is not always possible to interrupt read method.
726

730         }
731         if (addArgs != null) {
732             for (int i = 0; i != addArgs.length; ++i) {
733                 cmd[L + i] = ScriptRuntime.toString(addArgs[i]);
734             }
735         }
736

738         if (outBytes != null) {
739             String s = ScriptRuntime.toString(outObj) + outBytes.toString();

741         }
742         if (errBytes != null) {
743             String s = ScriptRuntime.toString(errObj) + errBytes.toString();
744             ScriptableObject.putProperty(params, "err", s);
745         }
746 
747         return new Integer(exitCode);

891     /**
892      * Runs the given process using Runtime.exec().
893      * If any of in, out, err is null, the corresponding process stream will
894      * be closed immediately, otherwise it will be closed as soon as
895      * all data will be read from/written to process
896      *
897      * @return Exit value of process.
898      * @throws IOException If there was an error executing the process.
899      */
900     private static int runProcess(String[] cmd, String[] environment,
901                                   InputStream in, OutputStream out,
902                                   OutputStream err)
903         throws IOException
904     {
905         Process p;
906         if (environment == null) {
907             p = Runtime.getRuntime().exec(cmd);
908         } else {
909             p = Runtime.getRuntime().exec(cmd, environment);
910         }
911 
912         try {
913             PipeThread inThread = null;
914             if (in != null) {
915                 inThread = new PipeThread(false, in, p.getOutputStream());
916                 inThread.start();
917             } else {
918                 p.getOutputStream().close();
919             }
920 
921             PipeThread outThread = null;
922             if (out != null) {
923                 outThread = new PipeThread(true, p.getInputStream(), out);
924                 outThread.start();
925             } else {
926                 p.getInputStream().close();
927             }
928 
929             PipeThread errThread = null;
930             if (err != null) {
931                 errThread = new PipeThread(true, p.getErrorStream(), err);
932                 errThread.start();
933             } else {
934                 p.getErrorStream().close();
935             }
936 
937             // wait for process completion
938             for (;;) {
939                 try {
940                     p.waitFor();
941                     if (outThread != null) {
942                         outThread.join();
943                     }
944                     if (inThread != null) {
945                         inThread.join();
946                     }
947                     if (errThread != null) {
948                         errThread.join();
949                     }
950                     break;
951                 } catch (InterruptedException ignore) {
952                 }
953             }
954 
955             return p.exitValue();
956         } finally {
957             p.destroy();
958         }
959     }

961     static void pipe(boolean fromProcess, InputStream from, OutputStream to)
962         throws IOException
963     {
964         try {
965             final int SIZE = 4096;
966             byte[] buffer = new byte[SIZE];
967             for (;;) {
968                 int n;
969                 if (!fromProcess) {
970                     n = from.read(buffer, 0, SIZE);
971                 } else {
972                     try {
973                         n = from.read(buffer, 0, SIZE);
974                     } catch (IOException ex) {
975                         // Ignore exception as it can be cause by closed pipe
976                         break;
977                     }
978                 }
979                 if (n < 0) { break; }
980                 if (fromProcess) {
981                     to.write(buffer, 0, n);
982                     to.flush();
983                 } else {
984                     try {
985                         to.write(buffer, 0, n);
986                         to.flush();
987                     } catch (IOException ex) {
988                         // Ignore exception as it can be cause by closed pipe
989                         break;
990                     }
991                 }
992             }
993         } finally {
994             try {
995                 if (fromProcess) {
996                     from.close();
997                 } else {
998                     to.close();
999                 }
1000             } catch (IOException ex) {
1001                 // Ignore errors on close. On Windows JVM may throw invalid
1002                 // refrence exception if process terminates too fast.
1003             }
1004         }
1005     }

1043                                   boolean urlIsFile)
1044         throws IOException
1045     {
1046         int chunkLength;
1047         InputStream is = null;

1051                 URLConnection uc = urlObj.openConnection();

1059                         charCoding = getCharCodingFromType(type);
1060                     }

1064                 if (!f.exists()) {
1065                     throw new FileNotFoundException("File not found: " + filePath);
1066                 } else if (!f.canRead()) {
1067                     throw new IOException("Cannot read file: " + filePath);
1068                 }
1069                 long length = f.length();
1070                 chunkLength = (int)length;
1071                 if (chunkLength != length)
1072                     throw new IOException("Too big file size: "+length);
1073 
1074                 if (chunkLength == 0) { return ""; }
1075 
1076                 is = new FileInputStream(f);
1077             }
1078 
1079             Reader r;
1080             if (charCoding == null) {
1081                 r = new InputStreamReader(is);
1082             } else {
1083                 r = new InputStreamReader(is, charCoding);
1084             }
1085             return readReader(r, chunkLength);
1086 
1087         } finally {
1088             if (is != null)

1094     {
1095         int i = type.indexOf(';');
1096         if (i >= 0) {
1097             int end = type.length();
1098             ++i;
1099             while (i != end && type.charAt(i) <= ' ') {
1100                 ++i;
1101             }
1102             String charset = "charset";
1103             if (charset.regionMatches(true, 0, type, i, charset.length()))
1104             {
1105                 i += charset.length();
1106                 while (i != end && type.charAt(i) <= ' ') {
1107                     ++i;
1108                 }
1109                 if (i != end && type.charAt(i) == '=') {
1110                     ++i;
1111                     while (i != end && type.charAt(i) <= ' ') {
1112                         ++i;
1113                     }
1114                     if (i != end) {
1115                         // i is at the start of non-empty
1116                         // charCoding spec
1117                         while (type.charAt(end -1) <= ' ') {
1118                             --end;
1119                         }
1120                         return type.substring(i, end);
1121                     }
1122                 }
1123             }
1124         }
1125         return null;

1134     private static String readReader(Reader reader, int initialBufferSize)
1135         throws IOException
1136     {
1137         char[] buffer = new char[initialBufferSize];
1138         int offset = 0;

1141             if (n < 0) { break;    }
1142             offset += n;

1145                 System.arraycopy(buffer, 0, tmp, 0, offset);

34  package org.mozilla.javascript.tools.debugger.treetable;
35  
36  import javax.swing.tree.TreeModel;
37  
38  /**
39   * TreeTableModel is the model used by a JTreeTable. It extends TreeModel
40   * to add methods for getting inforamtion about the set of columns each
41   * node in the TreeTableModel may have. Each column, like a column in
42   * a TableModel, has a name and a type associated with it. Each node in
43   * the TreeTableModel can return a value for each of the columns and
44   * set that value if isCellEditable() returns true.
45   *
46   * @author Philip Milne
47   * @author Scott Violet
48   */
49  public interface TreeTableModel extends TreeModel
50  {
51      /**
52       * Returns the number ofs availible column.
53       */
54      public int getColumnCount();
55  
56      /**
57       * Returns the name for column number <code>column</code>.
58       */
59      public String getColumnName(int column);
60  
61      /**
62       * Returns the type for column number <code>column</code>.
63       */
64      public Class<?> getColumnClass(int column);
65  
66      /**
67       * Returns the value to be displayed for node <code>node</code>,
68       * at column number <code>column</code>.
69       */
70      public Object getValueAt(Object node, int column);
71  
72      /**
73       * Indicates whether the the value for node <code>node</code>,
74       * at column number <code>column</code> is editable.
75       */
76      public boolean isCellEditable(Object node, int column);
77  
78      /**
79       * Sets the value for node <code>node</code>,
80       * at column number <code>column</code>.
81       */
82      public void setValueAt(Object aValue, Object node, int column);
83  }

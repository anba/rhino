233     /**
234      * Returns a name for the token.  If Rhino is compiled with certain
235      * hardcoded debugging flags in this file, it calls {@code #typeToName};
236      * otherwise it returns a string whose value is the token number.
237      */
238     public static String name(int token)
239     {
240         if (!printNames) {
241             return String.valueOf(token);
242         }
243         return typeToName(token);
244     }

466     /**
467      * Return true if the passed code is a valid Token constant.
468      * @param code a potential token code
469      * @return true if it's a known token
470      */
471     public static boolean isValidToken(int code) {
472         return code >= ERROR
473                 && code <= LAST_TOKEN;
